%% 本文档系统阐述视频生成的核心技术、架构设计与工程实现，为AI开发人员提供从理论到实践的完整参考 %%

## 一、核心概念：视频生成的挑战与机遇

**视频生成**（Video Generation）是指利用深度学习模型**从条件输入（如文本、图像、音频）生成连续、连贯、高质量的视频序列**的技术。与图像生成相比，视频生成需要额外建模**时间维度**的连续性和一致性，这是其核心挑战所在。

### 1.1 视频生成的特殊性

#### 关键特征：
1. **时空联合建模**：同时处理空间（H×W）和时间（T）维度
2. **多尺度生成**：从粗到细的层次化生成策略
3. **条件控制多样性**：文本、图像、音频、动作等多种控制方式
4. **物理合理性**：符合现实世界的物理规律和运动学

### 2.1 GAN-based 视频生成

**代表工作**：VideoGAN、TGAN、StyleGAN-V

#### 架构特点：
```python
# GAN-based视频生成器
class VideoGANGenerator:
    def __init__(self, latent_dim=512, video_length=16):
        # 潜在空间映射
        self.mapping_network = MappingNetwork(latent_dim, 512)
        
        # 时空合成网络
        self.synthesis_network = SpatiotemporalSynthesisNetwork(
            resolution=256,
            num_frames=video_length
        )
        
        # 判别器
        self.discriminator = VideoDiscriminator(
            temporal_stride=2,
            spatial_stride=2
        )
    
    def generate(self, z, condition=None):
        """从潜在向量生成视频"""
        # 潜在向量映射
        w = self.mapping_network(z)
        
        # 条件融合
        if condition is not None:
            w = self._fuse_condition(w, condition)
        
        # 时空合成
        video = self.synthesis_network(w)
        
        return video  # [B, T, C, H, W]
    
    def adversarial_loss(self, real_videos, fake_videos):
        """计算对抗损失"""
        # 判别器前向
        real_scores = self.discriminator(real_videos)
        fake_scores = self.discriminator(fake_videos)
        
        # 生成器损失（最大化判别器对假视频的评分）
        g_loss = -torch.mean(fake_scores)
        
        # 判别器损失（区分真假）
        d_loss = torch.mean(fake_scores) - torch.mean(real_scores)
        
        # 梯度惩罚（WGAN-GP）
        gp = self._gradient_penalty(real_videos, fake_videos)
        d_loss += 10 * gp
        
        return g_loss, d_loss
```

#### 优势与局限：
- **优势**：训练相对稳定、生成速度快
- **局限**：模式崩溃、训练不稳定、视频长度有限

### 2.2 自回归视频生成

**代表工作**：VideoGPT、DVD-GAN

#### 架构特点：
```python
# 自回归视频生成器
class AutoregressiveVideoGenerator:
    def __init__(self, vocab_size=1024, hidden_dim=768):
        # 视频Tokenizer（VQ-VAE）
        self.tokenizer = VideoTokenizer(
            codebook_size=vocab_size,
            spatial_downsample=4,
            temporal_downsample=2
        )
        
        # 自回归Transformer
        self.transformer = VideoTransformer(
            hidden_dim=hidden_dim,
            num_layers=24,
            num_heads=12,
            max_seq_len=4096
        )
        
        # 解码器
        self.decoder = VideoDecoder(
            hidden_dim=hidden_dim,
            vocab_size=vocab_size
        )
    
    def generate(self, condition_tokens, max_length=1024):
        """自回归生成视频token"""
        generated_tokens = condition_tokens.clone()
        
        for i in range(max_length - len(condition_tokens)):
            # 前向传播
            logits = self.transformer(generated_tokens)
            next_token_logits = logits[:, -1, :]
            
            # 采样下一个token
            next_token = torch.multinomial(
                F.softmax(next_token_logits, dim=-1), 
                num_samples=1
            )
            
            # 添加到序列
            generated_tokens = torch.cat([generated_tokens, next_token], dim=1)
            
            # 停止条件（遇到EOS token）
            if next_token.item() == self.tokenizer.eos_token_id:
                break
        
        # 解码为视频
        video = self.tokenizer.decode(generated_tokens)
        return video
```

#### 核心创新：
1. **离散表示**：使用VQ-VAE将视频压缩为离散token
2. **序列建模**：将视频生成视为序列预测问题
3. **条件生成**：支持多种条件输入（文本、类别等）

### 2.3 扩散模型视频生成

**代表工作**：[[Video Diffusion]]、[[Stable Video Diffusion]]、[[CogVideo]]

#### 架构特点：
```python
# 视频扩散模型
class VideoDiffusionModel:
    def __init__(self, model_type="3d_unet"):
        # 噪声调度器
        self.noise_scheduler = NoiseScheduler(
            beta_start=0.0001,
            beta_end=0.02,
            num_timesteps=1000
        )
        
        # 去噪网络
        if model_type == "3d_unet":
            self.denoiser = SpatiotemporalUNet(
                in_channels=4,  # RGB + 时间
                out_channels=4,
                num_res_blocks=2,
                attention_resolutions=[16, 8],
                channel_mult=[1, 2, 4, 4],
                num_heads=8,
                use_spatial_transformer=True,
                use_temporal_transformer=True
            )
        elif model_type == "transformer":
            self.denoiser = VideoTransformerDenoiser(
                hidden_dim=1024,
                num_layers=24,
                num_heads=16,
                patch_size=[1, 2, 2]  # [T, H, W]
            )
        
        # 条件编码器
        self.condition_encoder = ConditionEncoder(
            text_dim=768,
            image_dim=512,
            audio_dim=256
        )
    
    def forward_diffusion(self, x_0, t):
        """前向扩散过程"""
        # 采样噪声
        noise = torch.randn_like(x_0)
        
        # 计算噪声系数
        sqrt_alpha_cumprod = self.noise_scheduler.sqrt_alpha_cumprod[t]
        sqrt_one_minus_alpha_cumprod = self.noise_scheduler.sqrt_one_minus_alpha_cumprod[t]
        
        # 添加噪声
        x_t = sqrt_alpha_cumprod * x_0 + sqrt_one_minus_alpha_cumprod * noise
        
        return x_t, noise
    
    def reverse_diffusion(self, x_t, t, condition=None):
        """反向去噪过程"""
        # 条件编码
        if condition is not None:
            cond_emb = self.condition_encoder(condition)
        else:
            cond_emb = None
        
        # 时间步嵌入
        t_emb = self._get_timestep_embedding(t)
        
        # 去噪预测
        predicted_noise = self.denoiser(x_t, t_emb, cond_emb)
        
        return predicted_noise
    
    def training_loss(self, x_0, condition=None):
        """训练损失计算"""
        # 随机采样时间步
        t = torch.randint(0, self.noise_scheduler.num_timesteps, (x_0.shape[0],))
        
        # 前向扩散
        x_t, noise = self.forward_diffusion(x_0, t)
        
        # 预测噪声
        predicted_noise = self.reverse_diffusion(x_t, t, condition)
        
        # 噪声预测损失
        loss = F.mse_loss(predicted_noise, noise)
        
        return loss
```

#### 扩散模型优势：
1. **训练稳定性**：避免GAN的模式崩溃问题
2. **生成质量**：通常能产生更高质量、更多样的样本
3. **条件控制**：灵活支持多种条件输入方式
4. **渐进生成**：从噪声逐步生成，支持中间编辑

## 三、核心架构：时空建模的技术实现

%% 时空建模是视频生成的核心，决定了模型对动态内容的理解和生成能力 %%

### 3.1 时空注意力机制

**核心思想**：将传统的2D空间注意力扩展为3D时空注意力，同时建模空间和时间关系。

```python
# 时空注意力实现
class SpatiotemporalAttention(nn.Module):
    def __init__(self, dim, num_heads=8, qkv_bias=False):
        super().__init__()
        self.num_heads = num_heads
        self.head_dim = dim // num_heads
        self.scale = self.head_dim ** -0.5
        
        # 投影层
        self.qkv = nn.Linear(dim, dim * 3, bias=qkv_bias)
        self.proj = nn.Linear(dim, dim)
        
        # 相对位置编码
        self.rel_pos_embed = RelativePositionEmbedding3D(
            spatial_size=32,
            temporal_size=16,
            dim=dim
        )
    
    def forward(self, x, mask=None):
        """时空注意力计算"""
        B, T, H, W, C = x.shape
        N = T * H * W  # 总token数
        
        # 重塑为序列
        x_flat = x.reshape(B, N, C)
        
        # 计算QKV
        qkv = self.qkv(x_flat).reshape(B, N, 3, self.num_heads, self.head_dim)
        qkv = qkv.permute(2, 0, 3, 1, 4)  # [3, B, num_heads, N, head_dim]
        q, k, v = qkv[0], qkv[1], qkv[2]
        
        # 注意力计算
        attn = (q @ k.transpose(-2, -1)) * self.scale
        
        # 添加相对位置偏置
        rel_pos_bias = self.rel_pos_embed(q.shape[2], k.shape[2])
        attn = attn + rel_pos_bias
        
        # 应用掩码（如果有）
        if mask is not None:
            attn = attn.masked_fill(mask == 0, float('-inf'))
        
        # Softmax和值加权
        attn = attn.softmax(dim=-1)
        x = (attn @ v).transpose(1, 2).reshape(B, N, C)
        
        # 投影输出
        x = self.proj(x)
        
        # 重塑回原始形状
        x = x.reshape(B, T, H, W, C)
        
        return x
```

#### 3.1.1 分解时空注意力

为了降低计算复杂度，常采用分解的注意力机制：

```python
# 分解的时空注意力
class FactorizedSpatiotemporalAttention(nn.Module):
    def __init__(self, dim, num_heads=8):
        super().__init__()
        # 空间注意力
        self.spatial_attention = nn.MultiheadAttention(
            embed_dim=dim,
            num_heads=num_heads,
            batch_first=True
        )
        
        # 时间注意力
        self.temporal_attention = nn.MultiheadAttention(
            embed_dim=dim,
            num_heads=num_heads,
            batch_first=True
        )
        
        # 层归一化
        self.norm1 = nn.LayerNorm(dim)
        self.norm2 = nn.LayerNorm(dim)
    
    def forward(self, x):
        """分解的时空注意力"""
        B, T, H, W, C = x.shape
        
        # 空间注意力（每个时间步独立）
        # 重塑: [B, T, H*W, C]
        spatial_tokens = x.reshape(B, T, H*W, C)
        
        for t in range(T):
            spatial_out, _ = self.spatial_attention(
                query=spatial_tokens[:, t, :, :],
                key=spatial_tokens[:, t, :, :],
                value=spatial_tokens[:, t, :, :]
            )
            spatial_tokens[:, t, :, :] = self.norm1(spatial_out)
        
        # 时间注意力（每个空间位置独立）
        # 重塑: [B, H*W, T, C]
        temporal_tokens = spatial_tokens.transpose(1, 2)
        
        for hw in range(H*W):
            temporal_out, _ = self.temporal_attention(
                query=temporal_tokens[:, hw, :, :],
                key=temporal_tokens[:, hw, :, :],
                value=temporal_tokens[:, hw, :, :]
            )
            temporal_tokens[:, hw, :, :] = self.norm2(temporal_out)
        
        # 重塑回原始形状
        output = temporal_tokens.transpose(1, 2).reshape(B, T, H, W, C)
        
        return output
```

### 3.2 3D U-Net 架构

**代表模型**：[[Stable Video Diffusion]]、[[Video Diffusion]]

#### 架构设计：
```python
# 3D U-Net视频生成器
class SpatiotemporalUNet(nn.Module):
    def __init__(self, in_channels=4, out_channels=4):
        super().__init__()
        
        # 编码器（下采样）
        self.encoder = nn.ModuleList([
            DownsampleBlock(in_channels, 64),
            DownsampleBlock(64, 128),
            DownsampleBlock(128, 256),
            DownsampleBlock(256, 512),
            DownsampleBlock(512, 512)  # 瓶颈层
        ])
        
        # 中间层（时空Transformer）
        self.middle_block = SpatiotemporalTransformerBlock(
            dim=512,
            num_heads=8,
            num_layers=2
        )
        
        # 解码器（上采样）
        self.decoder = nn.ModuleList([
            UpsampleBlock(512, 512, skip_channels=512),
            UpsampleBlock(1024, 256, skip_channels=256),  # 1024 = 512 + 512
            UpsampleBlock(512, 128, skip_channels=128),   # 512 = 256 + 256
            UpsampleBlock(256, 64, skip_channels=64),     # 256 = 128 + 128
            UpsampleBlock(128, 64, skip_channels=64)      # 128 = 64 + 64
        ])
        
        # 输出层
        self.output_conv = nn.Conv3d(64, out_channels, kernel_size=3, padding=1)
        
        # 时间步嵌入
        self.time_embed = TimestepEmbedding(512)
        
        # 条件嵌入
        self.cond_embed = ConditionEmbedding(768, 512)  # 文本条件
    
    def forward(self, x, t, cond=None):
        """前向传播"""
        # 时间步嵌入
        t_emb = self.time_embed(t)
        
        # 条件嵌入
        if cond is not None:
            cond_emb = self.cond_embed(cond)
            # 融合时间和条件
            emb = t_emb + cond_emb
        else:
            emb = t_emb
        
        # 编码器前向
        skips = []
        for block in self.encoder:
            x = block(x, emb)
            skips.append(x)
        
        # 中间层
        x = self.middle_block(x, emb)
        
        # 解码器前向（带跳跃连接）
        for i, block in enumerate(self.decoder):
            skip = skips[-(i+1)]
            x = block(x, skip, emb)
        
        # 输出
        output = self.output_conv(x)
        
        return output
```

#### 3D卷积 vs 2D卷积+时间注意力：

```python
# 3D卷积与2D+时间注意力对比
class SpatiotemporalModelingStrategies:
    @staticmethod
    def pure_3d_convolution(x, kernel_size=(3, 3, 3)):
        """纯3D卷积建模"""
        # x: [B, C, T, H, W]
        conv3d = nn.Conv3d(
            in_channels=x.shape[1],
            out_channels=x.shape[1],
            kernel_size=kernel_size,
            padding=tuple(k//2 for k in kernel_size)
        )
        return conv3d(x)
    
    @staticmethod
    def separable_3d_convolution(x):
        """分离的3D卷积（2D空间+1D时间）"""
        # 空间卷积（每个时间步独立）
        spatial_conv = nn.Conv2d(
            in_channels=x.shape[1],
            out_channels=x.shape[1],
            kernel_size=3,
            padding=1
        )
        
        # 时间卷积（每个空间位置独立）
        temporal_conv = nn.Conv1d(
            in_channels=x.shape[1],
            out_channels=x.shape[1],
            kernel_size=3,
            padding=1
        )
        
        # 应用空间卷积
        B, C, T, H, W = x.shape
        spatial_out = torch.stack([
            spatial_conv(x[:, :, t, :, :]) for t in range(T)
        ], dim=2)
        
        # 应用时间卷积
        # 重塑: [B, C, H*W, T]
        temporal_input = spatial_out.reshape(B, C, T, H*W).transpose(2, 3)
        temporal_out = temporal_conv(temporal_input)
        
        # 重塑回原始形状
        output = temporal_out.transpose(2, 3).reshape(B, C, T, H, W)
        
        return output
    
    @staticmethod
    def conv2d_plus_temporal_attention(x, num_heads=8):
        """2D卷积+时间注意力"""
        # 2D空间卷积
        spatial_conv = nn.Conv2d(
            in_channels=x.shape[1],
            out_channels=x.shape[1],
            kernel_size=3,
            padding=1
        )
        
        # 时间注意力
        temporal_attention = nn.MultiheadAttention(
            embed_dim=x.shape[1],
            num_heads=num_heads,
            batch_first=True
        )
        
        # 应用空间卷积
        B, C, T, H, W = x.shape
        spatial_out = torch.stack([
            spatial_conv(x[:, :, t, :, :]) for t in range(T)
        ], dim=2)
        
        # 应用时间注意力
        # 重塑: [B, H*W, T, C]
        attention_input = spatial_out.reshape(B, C, T, H*W)
        attention_input = attention_input.permute(0, 3, 2, 1)  # [B, H*W, T, C]
        
        # 展平空间维度
        flat_input = attention_input.reshape(B*H*W, T, C)
        
        # 时间注意力
        attention_output, _ = temporal_attention(flat_input, flat_input, flat_input)
        
        # 重塑回原始形状
        output = attention_output.reshape(B, H*W, T, C)
        output = output.permute(0, 3, 2, 1).reshape(B, C, T, H, W)
        
        return output
```

```python
class VideoTransformerGenerator(nn.Module):
    def __init__(self, patch_size=(1, 2, 2), hidden_dim=1024):
        super().__init__()
        self.patch_size = patch_size

            embed_dim=hidden_dim
        )
        
        # Transformer编码器
        self.transformer = nn.TransformerEncoder(

                nhead=16,
                dim_feedforward=4096,
                dropout=0.1,
                batch_first=True
            ),
            num_layers=24
        )
        
        # 条件编码器
        self.condition_encoder = MultimodalConditionEncoder(
            text_dim=768,
            image_dim=512,
            hidden_dim=hidden_dim
        )
        
        # 解码器（Patch到视频）
        self.decoder = VideoPatchDecoder(
            patch_size=patch_size,
            embed_dim=hidden_dim,
            out_channels=3
        )
    
    def forward(self, video_tokens, condition=None):
        """前向传播"""
        # 视频token嵌入
        token_embeddings = self.patch_embed(video_tokens)
        
        # 添加位置编码
        pos_embeddings = self.pos_embed(token_embeddings.shape[1])
        x = token_embeddings + pos_embeddings
        
        # 条件编码
        if condition is not None:
            cond_emb = self.condition_encoder(condition)
            # 将条件作为前缀token添加
            x = torch.cat([cond_emb.unsqueeze(1), x], dim=1)
        
        # Transformer编码
        transformer_output = self.transformer(x)
        
        # 解码为视频
        if condition is not None:
            # 移除条件token
            transformer_output = transformer_output[:, 1:, :]
        
        video_output = self.decoder(transformer_output)
        
        return video_output
```

```python
# 视频Patch处理
class VideoPatchProcessor:
    def __init__(self, patch_size=(1, 2, 2)):
        self.patch_size = patch_size  # [T, H, W]
    
    def video_to_patches(self, video):
        """将视频转换为patch序列"""
        B, C, T, H, W = video.shape
        pt, ph, pw = self.patch_size
        
        # 确保尺寸可被patch大小整除
        assert T % pt == 0 and H % ph == 0 and W % pw == 0
        
        # 重塑为patch
        patches = video.unfold(2, pt, pt).unfold(3, ph, ph).unfold(4, pw, pw)
        # patches形状: [B, C, T/pt, H/ph, W/pw, pt, ph, pw]
        
        # 展平
        patches = patches.permute(0, 2, 3, 4, 1, 5, 6, 7)  # [B, T', H', W', C, pt, ph, pw]
        patches = patches.reshape(B, -1, C * pt * ph * pw)  # [B, num_patches, patch_dim]
        
        return patches
    
    def patches_to_video(self, patches, original_shape):
        """将patch序列转换回视频"""
        B, num_patches, patch_dim = patches.shape
        C, T, H, W = original_shape[1], original_shape[2], original_shape[3], original_shape[4]
        pt, ph, pw = self.patch_size
        
        # 计算patch网格大小
        T_patches = T // pt
        H_patches = H // ph
        W_patches = W // pw
        
        # 重塑为网格
        patches = patches.reshape(B, T_patches, H_patches, W_patches, C, pt, ph, pw)
        patches = patches.permute(0, 4, 1, 5, 2, 6, 3, 7)  # [B, C, T', pt, H', ph, W', pw]
        
        # 合并patch
        video = patches.reshape(B, C, T, H, W)
        
        return video
```

```python
    def __init__(self, text_encoder="CLIP", hidden_dim=768):

            self.text_encoder = CLIPTextEncoder()

            max_frames=128,

        text_embeddings = self.text_encoder(text_prompts)  # [B, L, D_text]

        aligned_embeddings = self.alignment_module(text_embeddings)
        
        # 时间扩展（为每个帧生成条件）
        temporal_conditions = self.temporal_extender(aligned_embeddings, num_frames)
        # temporal_conditions: [B, num_frames, D_video]

    def compute_alignment_loss(self, generated_video, text_prompts):

        # 提取视频特征
        video_features = self._extract_video_features(generated_video)
        
        # 文本特征
        text_features = self.text_encoder(text_prompts)
        
        # 对比学习损失（CLIP风格）
        loss = self._contrastive_loss(video_features, text_features)
        
        return loss
```

```python
# 图像到视频生成器
class ImageToVideoGenerator:
    def __init__(self):
        # 图像编码器
        self.image_encoder = ImageEncoder(
            backbone="ViT-L/14",
            output_dim=1024
        )
        
        # 运动预测网络
        self.motion_predictor = MotionPredictionNetwork(
            hidden_dim=1024,
            num_motion_codes=64
        )
        
        # 视频生成器
        self.video_generator = VideoGenerationNetwork(
            condition_dim=1024,
            motion_dim=256
        )
    
    def generate(self, image, motion_description=None, num_frames=16):
        """从图像生成视频"""
        # 编码输入图像
        image_features = self.image_encoder(image)  # [B, D]
        
        # 预测运动
        if motion_description is not None:
            # 使用文本描述指导运动
            motion_features = self._encode_motion_description(motion_description)
        else:
            # 自动预测合理运动
            motion_features = self.motion_predictor(image_features)
        
        # 扩展为时间序列
        temporal_conditions = self._create_temporal_sequence(
            image_features, motion_features, num_frames
        )
        
        # 生成视频
        video = self.video_generator(temporal_conditions)
        
        return video
    
    def _create_temporal_sequence(self, image_features, motion_features, num_frames):
        """创建时间序列条件"""
        # 基础图像特征
        base_condition = image_features.unsqueeze(1)  # [B, 1, D]
        
        # 运动特征（随时间变化）
        time_steps = torch.arange(num_frames).float() / num_frames
        time_embeddings = self._get_time_embeddings(time_steps)  # [num_frames, D_t]
        
        # 融合运动和时间
        motion_sequence = motion_features.unsqueeze(1) * time_embeddings.unsqueeze(0)
        
        # 结合基础条件和运动
        temporal_conditions = base_condition + motion_sequence
        
        return temporal_conditions  # [B, num_frames, D]
```

    def depth_guided_control(image, depth_map, depth_scale=1.0):
        """深度引导控制"""
        # depth_map: [H, W] 深度图（0-1范围）

        depth_analysis = {
            "min_depth": np.min(depth_map),
            "max_depth": np.max(depth_map),
            "depth_range": np.max(depth_map) - np.min(depth_map),
            "depth_layers": ImageToVideoMotionControl._segment_depth_layers(depth_map)
        }
        
        # 深度引导参数
        depth_params = {
            "depth_map": depth_map,
            "scale": depth_scale,
            "parallax_effect": True,  # 视差效果
            "focus_plane": 0.5,  # 焦点平面
            "depth_of_field": 0.3  # 景深
        }
        
        # 生成深度引导的运动场
        motion_field = ImageToVideoMotionControl._generate_depth_motion_field(
            depth_map, depth_params
        )
        
        return {
            "depth_analysis": depth_analysis,
            "depth_params": depth_params,
            "motion_field": motion_field
        }
    
    @staticmethod
    def _segment_depth_layers(depth_map):
        """分割深度层"""
        # 使用K-means聚类分割深度层
        depth_values = depth_map.flatten()
        kmeans = KMeans(n_clusters=3, random_state=42)
        labels = kmeans.fit_predict(depth_values.reshape(-1, 1))
        
        layers = []
        for i in range(3):
            layer_mask = (labels == i).reshape(depth_map.shape)
            layer_depth = np.mean(depth_values[labels == i])
            layers.append({
                "layer_id": i,
                "depth_value": float(layer_depth),
                "mask": layer_mask,
                "area_ratio": np.sum(layer_mask) / depth_map.size
            })
        
        return layers
    
    @staticmethod
    def _generate_depth_motion_field(depth_map, depth_params):
        """生成深度运动场"""
        h, w = depth_map.shape
        
        # 基于深度的运动向量
        motion_field = np.zeros((h, w, 2))  # [dx, dy]
        
        for i in range(h):
            for j in range(w):
                depth = depth_map[i, j]
                
                # 深度越近，运动幅度越大（视差效应）
                motion_scale = 1.0 - depth  # 近处物体运动更大
                
                # 相机运动模拟
                if depth_params["parallax_effect"]:
                    # 水平视差
                    dx = motion_scale * 0.1 * (j - w/2) / (w/2)
                    # 垂直视差
                    dy = motion_scale * 0.05 * (i - h/2) / (h/2)
                else:
                    dx = dy = 0
                
                motion_field[i, j, 0] = dx
                motion_field[i, j, 1] = dy
        
        return motion_field

### 5.2.1 时序损失函数设计

```python
class TemporalConsistencyLoss:
    """时序一致性损失函数"""
    
    def __init__(self):
        self.loss_components = {
            "optical_flow": OpticalFlowConsistencyLoss(),
            "feature_similarity": FeatureSimilarityLoss(),
            "temporal_smoothness": TemporalSmoothnessLoss(),
            "cycle_consistency": CycleConsistencyLoss()
        }
    
    def compute_loss(self, video_frames, gt_frames=None):
        """计算时序一致性损失"""
        losses = {}
        
        # 光流一致性损失
        if "optical_flow" in self.loss_components:
            flow_loss = self.loss_components["optical_flow"](
                video_frames, gt_frames
            )
            losses["optical_flow"] = flow_loss
        
        # 特征相似性损失
        if "feature_similarity" in self.loss_components:
            feature_loss = self.loss_components["feature_similarity"](
                video_frames
            )
            losses["feature_similarity"] = feature_loss
        
        # 时序平滑性损失
        if "temporal_smoothness" in self.loss_components:
            smoothness_loss = self.loss_components["temporal_smoothness"](
                video_frames
            )
            losses["temporal_smoothness"] = smoothness_loss
        
        # 循环一致性损失
        if "cycle_consistency" in self.loss_components:
            cycle_loss = self.loss_components["cycle_consistency"](
                video_frames
            )
            losses["cycle_consistency"] = cycle_loss
        
        # 加权总损失
        total_loss = sum(
            loss * weight for loss, weight in zip(
                losses.values(), 
                [0.4, 0.3, 0.2, 0.1]  # 权重配置
            )
        )
        
        return total_loss, losses


class OpticalFlowConsistencyLoss:
    """光流一致性损失"""
    
    def __call__(self, generated_frames, gt_frames=None):
        """计算光流一致性"""
        batch_size, num_frames, h, w, c = generated_frames.shape
        
        total_loss = 0
        for b in range(batch_size):
            for t in range(num_frames - 1):
                # 计算相邻帧之间的光流
                flow_forward = self._compute_optical_flow(
                    generated_frames[b, t],
                    generated_frames[b, t + 1]
                )
                
                # 如果有真实帧，计算真实光流
                if gt_frames is not None:
                    gt_flow = self._compute_optical_flow(
                        gt_frames[b, t],
                        gt_frames[b, t + 1]
                    )
                    # 光流差异损失
                    flow_diff = torch.abs(flow_forward - gt_flow).mean()
                    total_loss += flow_diff
                else:
                    # 自监督：warp前一帧并与当前帧比较
                    warped_frame = self._warp_frame(
                        generated_frames[b, t],
                        flow_forward
                    )
                    recon_loss = F.mse_loss(
                        warped_frame, 
                        generated_frames[b, t + 1]
                    )
                    total_loss += recon_loss
        
        return total_loss / (batch_size * (num_frames - 1))
```

```python
class TemporalAttentionOptimization:
    """时序注意力优化"""
    
    def __init__(self, attention_heads=8, temporal_window=5):
        self.attention_heads = attention_heads
        self.temporal_window = temporal_window
        
    def optimize_temporal_attention(self, attention_weights):
        """优化时序注意力权重"""
        optimized_weights = attention_weights.clone()
        
        # 1. 增强局部时序注意力
        optimized_weights = self._enhance_local_attention(
            optimized_weights
        )
        
        # 2. 抑制长距离噪声注意力
        optimized_weights = self._suppress_noisy_attention(
            optimized_weights
        )
        
        # 3. 确保注意力归一化
        optimized_weights = self._normalize_attention(
            optimized_weights
        )
        
        return optimized_weights
    
    def _enhance_local_attention(self, attention_weights):
        """增强局部时序注意力"""
        # 创建局部注意力掩码
        local_mask = self._create_local_mask(
            attention_weights.shape[-1],  # 序列长度
            self.temporal_window
        )
        
        # 增强局部注意力
        enhanced_weights = attention_weights * local_mask
        enhanced_weights = enhanced_weights * 1.5  # 局部注意力增强因子
        
        return enhanced_weights
    
    def _suppress_noisy_attention(self, attention_weights):
        """抑制噪声注意力"""
        # 计算注意力熵
        attention_entropy = self._compute_attention_entropy(
            attention_weights
        )
        
        # 高熵表示注意力分散，需要抑制
        entropy_mask = (attention_entropy > 0.8).float()
        suppressed_weights = attention_weights * (1 - entropy_mask * 0.3)
        
        return suppressed_weights
    
    def _create_local_mask(self, seq_len, window_size):
        """创建局部注意力掩码"""
        mask = torch.zeros(seq_len, seq_len)
        
        for i in range(seq_len):
            start = max(0, i - window_size // 2)
            end = min(seq_len, i + window_size // 2 + 1)
            mask[i, start:end] = 1.0
        
        return mask
```

```python
class FrameWiseTraining:
    """分帧训练策略"""
    
    def __init__(self, frame_window=16, overlap=4):
        self.frame_window = frame_window
        self.overlap = overlap
        
    def train_with_sliding_window(self, model, long_video):
        """使用滑动窗口训练长视频"""
        batch_size, total_frames, h, w, c = long_video.shape
        
        # 创建滑动窗口
        windows = self._create_sliding_windows(
            total_frames, self.frame_window, self.overlap
        )
        
        losses = []
        for window_start, window_end in windows:
            # 提取窗口内的帧
            window_frames = long_video[:, window_start:window_end]
            
            # 训练当前窗口
            loss = self._train_window(model, window_frames)
            losses.append(loss)
            
            # 更新模型参数
            self._update_model_parameters(model)
        
        return losses
    
    def _create_sliding_windows(self, total_frames, window_size, overlap):
        """创建滑动窗口"""
        windows = []
        step = window_size - overlap
        
        for start in range(0, total_frames - window_size + 1, step):
            end = start + window_size
            windows.append((start, end))
        
        return windows
    
    def inference_with_overlap(self, model, prompt, total_frames=64):
        """带重叠的推理"""
        generated_frames = []
        
        # 首先生成第一个窗口
        first_window = model.generate(
            prompt, 
            num_frames=self.frame_window
        )
        generated_frames.append(first_window)
        
        # 滑动生成后续窗口
        for i in range(1, total_frames // (self.frame_window - self.overlap)):
            # 使用重叠区域作为条件
            overlap_frames = generated_frames[-1][:, -self.overlap:]
            
            # 生成下一个窗口
            next_window = model.generate_conditioned(
                prompt,
                condition_frames=overlap_frames,
                num_frames=self.frame_window - self.overlap
            )
            
            # 拼接（去除重叠部分）
            generated_frames.append(next_window[:, self.overlap:])
        
        # 合并所有帧
        full_video = torch.cat(generated_frames, dim=1)
        return full_video
```

```python
class MultiScaleDiscriminator:
    """多尺度判别器"""
    
    def __init__(self, scales=[1, 0.5, 0.25]):
        self.scales = scales
        self.discriminators = nn.ModuleList()
        
        for scale in scales:
            discriminator = self._create_discriminator(scale)
            self.discriminators.append(discriminator)
    
    def _create_discriminator(self, scale):
        """创建指定尺度的判别器"""
        if scale == 1:
            return SpatialTemporalDiscriminator(resolution=256)
        elif scale == 0.5:
            return SpatialTemporalDiscriminator(resolution=128)
        elif scale == 0.25:
            return SpatialTemporalDiscriminator(resolution=64)
    
    def forward(self, video):
        """前向传播"""
        discriminator_outputs = []
        
        for scale, discriminator in zip(self.scales, self.discriminators):
            # 调整视频尺度
            scaled_video = F.interpolate(
                video, 
                scale_factor=scale,
                mode='bilinear',
                align_corners=False
            )
            
            # 判别器输出
            output = discriminator(scaled_video)
            discriminator_outputs.append(output)
        
        return discriminator_outputs
    
    def compute_loss(self, real_video, fake_video):
        """计算对抗损失"""
        real_outputs = self.forward(real_video)
        fake_outputs = self.forward(fake_video)
        
        losses = []
        for real_out, fake_out in zip(real_outputs, fake_outputs):
            # 判别器损失
            d_loss_real = F.binary_cross_entropy_with_logits(
                real_out, torch.ones_like(real_out)
            )
            d_loss_fake = F.binary_cross_entropy_with_logits(
                fake_out, torch.zeros_like(fake_out)
            )
            d_loss = (d_loss_real + d_loss_fake) / 2
            
            # 生成器损失
            g_loss = F.binary_cross_entropy_with_logits(
                fake_out, torch.ones_like(fake_out)
            )
            
            losses.append({
                "discriminator_loss": d_loss,
                "generator_loss": g_loss
            })
        
        return losses
```

```python
class TrainingStabilization:
    """训练稳定性技术"""
    
    def __init__(self):
        self.gradient_penalty_weight = 10.0
        self.spectral_norm = True
        self.gradient_clipping = True
        
    def apply_gradient_penalty(self, discriminator, real_data, fake_data):
        """应用梯度惩罚（WGAN-GP）"""
        batch_size = real_data.size(0)
        
        # 生成插值样本
        alpha = torch.rand(batch_size, 1, 1, 1, 1).to(real_data.device)
        interpolated = alpha * real_data + (1 - alpha) * fake_data
        interpolated.requires_grad_(True)
        
        # 计算判别器对插值样本的输出
        d_interpolated = discriminator(interpolated)
        
        # 计算梯度
        gradients = torch.autograd.grad(
            outputs=d_interpolated,
            inputs=interpolated,
            grad_outputs=torch.ones_like(d_interpolated),
            create_graph=True,
            retain_graph=True
        )[0]
        
        # 计算梯度惩罚
        gradients = gradients.view(batch_size, -1)
        gradient_norm = gradients.norm(2, dim=1)
        gradient_penalty = ((gradient_norm - 1) ** 2).mean()
        
        return gradient_penalty * self.gradient_penalty_weight
    
    def apply_spectral_norm(self, model):
        """应用谱归一化"""
        if self.spectral_norm:
            for module in model.modules():
                if isinstance(module, (nn.Conv2d, nn.Conv3d, nn.Linear)):
                    nn.utils.spectral_norm(module)
        
        return model
    
    def clip_gradients(self, model, max_norm=1.0):
        """梯度裁剪"""
        if self.gradient_clipping:
            torch.nn.utils.clip_grad_norm_(
                model.parameters(), 
                max_norm=max_norm
            )
```

```python
class EarlyStopping:
    """早停策略"""
    
    def __init__(self, patience=10, min_delta=0.001):
        self.patience = patience
        self.min_delta = min_delta
        self.counter = 0
        self.best_score = None
        self.early_stop = False
        
    def __call__(self, val_score):
        """检查是否应该早停"""
        if self.best_score is None:
            self.best_score = val_score
        elif val_score < self.best_score + self.min_delta:
            self.counter += 1
            if self.counter >= self.patience:
                self.early_stop = True
        else:
            self.best_score = val_score
            self.counter = 0
        
        return self.early_stop


class ModelSelection:
    """模型选择策略"""
    
    def select_best_model(self, model_checkpoints, val_metrics):
        """选择最佳模型"""
        # 综合多个指标
        composite_scores = []
        
        for checkpoint, metrics in zip(model_checkpoints, val_metrics):
            # 计算综合得分
            score = self._compute_composite_score(metrics)
            composite_scores.append((checkpoint, score))
        
        # 选择得分最高的模型
        best_checkpoint, best_score = max(
            composite_scores, 
            key=lambda x: x[1]
        )
        
        return best_checkpoint, best_score
    
    def _compute_composite_score(self, metrics):
        """计算综合得分"""
        weights = {
            "fvd": 0.3,      # Frechet Video Distance
            "psnr": 0.2,     # Peak Signal-to-Noise Ratio
            "ssim": 0.2,     # Structural Similarity
            "temporal_consistency": 0.15,
            "diversity": 0.15
        }
        
        score = 0
        for metric_name, weight in weights.items():
            if metric_name in metrics:
                # 归一化指标值
                normalized_value = self._normalize_metric(
                    metric_name, 
                    metrics[metric_name]
                )
                score += normalized_value * weight
        
        return score
```

```python
class VideoModelCompression:
    """视频模型压缩"""
    
    def __init__(self):
        self.compression_methods = {
            "pruning": {
                "types": ["unstructured", "structured", "global"],
                "sparsity_target": 0.5,
                "iterative": True
            },
            "quantization": {
                "precision": ["int8", "int4", "fp16", "bf16"],
                "calibration": "dynamic",
                "per_channel": True
            },
            "knowledge_distillation": {
                "teacher_model": "原始大模型",
                "student_model": "压缩小模型",
                "distillation_loss": "KL散度 + MSE"
            }
        }
    
    def apply_pruning(self, model, pruning_rate=0.3):
        """应用剪枝"""
        pruning_config = {
            "pruning_method": "l1_unstructured",
            "pruning_rate": pruning_rate,
            "iterative_steps": 3,
            "fine_tune_epochs": 10
        }
        
        # 创建剪枝器
        parameters_to_prune = []
        for name, module in model.named_modules():
            if isinstance(module, (nn.Conv2d, nn.Linear)):
                parameters_to_prune.append((module, 'weight'))
        
        # 应用迭代剪枝
        pruned_model = model
        for step in range(pruning_config["iterative_steps"]):
            current_rate = pruning_rate * (step + 1) / pruning_config["iterative_steps"]
            
            for module, param_name in parameters_to_prune:
                prune.l1_unstructured(module, param_name, amount=current_rate)
            
            # 微调
            if step < pruning_config["iterative_steps"] - 1:
                self._fine_tune(pruned_model, pruning_config["fine_tune_epochs"])
        
        # 移除剪枝掩码，使剪枝永久化
        for module, param_name in parameters_to_prune:
            prune.remove(module, param_name)
        
        return pruned_model
    
    def apply_quantization(self, model, quantization_type="int8"):
        """应用量化"""
        if quantization_type == "dynamic_int8":
            quantized_model = torch.quantization.quantize_dynamic(
                model,
                {nn.Linear, nn.Conv2d, nn.Conv3d},
                dtype=torch.qint8
            )
        elif quantization_type == "static_int8":
            # 需要校准
            model.eval()
            model.qconfig = torch.quantization.get_default_qconfig('fbgemm')
            torch.quantization.prepare(model, inplace=True)
            # 使用校准数据
            # torch.quantization.convert(model, inplace=True)
            quantized_model = model
        elif quantization_type == "fp16":
            quantized_model = model.half()
        else:
            raise ValueError(f"不支持的量化类型: {quantization_type}")
        
        return quantized_model
    
    def evaluate_compression_impact(self, original_model, compressed_model, test_data):
        """评估压缩影响"""
        evaluation_results = {
            "size_reduction": {
                "original_size": self._get_model_size(original_model),
                "compressed_size": self._get_model_size(compressed_model),
                "reduction_ratio": 0
            },
            "speed_improvement": {
                "original_inference_time": self._measure_inference_time(original_model, test_data),
                "compressed_inference_time": self._measure_inference_time(compressed_model, test_data),
                "speedup": 0
            },
            "quality_degradation": {
                "original_quality": self._evaluate_quality(original_model, test_data),
                "compressed_quality": self._evaluate_quality(compressed_model, test_data),
                "quality_drop": 0
            }
        }
        
        # 计算指标
        evaluation_results["size_reduction"]["reduction_ratio"] = (
            1 - evaluation_results["size_reduction"]["compressed_size"] / 
            evaluation_results["size_reduction"]["original_size"]
        )
        
        evaluation_results["speed_improvement"]["speedup"] = (
            evaluation_results["speed_improvement"]["original_inference_time"] /
            evaluation_results["speed_improvement"]["compressed_inference_time"]
        )
        
        return evaluation_results
```
