# 算子不兼容方案

## 概述

算子（Operator）不兼容是 ONNX 导出遇到的最棘手问题之一。当模型包含 ONNX 规范未定义或 ONNX Runtime 未实现的算子时，转换会失败或推理时报错。

**标签**: #unsupported-ops #custom-kernels #onnxruntime-extensions

---

## 诊断算子系统问题

### 场景1: 导出失败 - "Unsupported operator"

```python
import torch

class ComplexModel(torch.nn.Module):
    def forward(self, x):
        # 使用某些 PyTorch 特有算子
        result = torch.einsum('bij,bjk->bik', x, x.transpose(1, 2))  # Einsum
        # 或自定义算子
        return torch.custom_op(x)  # 用户自定义 C++ 算子

# 导出时直接失败
torch.onnx.export(
    model,
    x,
    "model.onnx",
    opset_version=13
)
# 报错: "RuntimeError: ONNX export failed: could not export python function..."
```

import onnxruntime as ort

#### 方法A: 检查 ONNX 模型中的所有算子

import onnx

def list_all_operators(model_path):
    """列出模型中的所有算子及其数量"""
    model = onnx.load(model_path)
    op_counts = {}

    for node in model.graph.node:
        op_type = node.op_type
        op_counts[op_type] = op_counts.get(op_type, 0) + 1

    for op_type, count in sorted(op_counts.items(), key=lambda x: x[1], reverse=True):
        print(f"  {op_type}: {count}")

## 常见不支持的算子

| 算子 | 状态 | 替代方案 | 备注 |
|------|------|---------|------|
| `Einsum` | ⚠️ 部分支持（opset≥12） | 矩阵运算链或 Reshape+MatMul | 使用 `torch.einsum` 可能导出为 `MatMul`+`Reshape` |
| `ScatterElements` | ✅ 支持（opset≥11） | 使用 `ScatterND` 或自定义 | ONNX Runtime 完全支持 |
| `ScatterND` | ⚠️ 部分支持 | 使用 `ScatterElements` 组合 | 复杂索引需要扩展 |
| `Unique` | ❌ 不支持 | 手动实现或预计算 | 少见，可替换为逻辑运算 |
| `CTCLoss` | ✅ 支持 | - | 需要特定 opset |
| `GridSample` | ⚠️ 实验性 | 使用 `onnxruntime-extensions` | ONNX Runtime 16.0+ |
| `FlashAttention` | ❌ 不支持 | 使用原生 Attention 实现 | 需要导出为标准 Attention |

## 解决方案策略

### 策略1: 导出时使用 custom_opsets

某些框架允许指定 `custom_opsets` 来映射不标准算子：

```python
import torch

# 方案: 修改模型避免特定算子
class ModelWithoutEinsum(torch.nn.Module):
    def __init__(self, original_model):
        super().__init__()
        self.original = original_model

    def forward(self, x):
        # 将 Einsum 替换为标准运算
        # 例如: torch.einsum('bij,bjk->bik', A, B)
        # 等价于: torch.matmul(A, B)
        return torch.matmul(x, x.transpose(1, 2))

# 或使用 exporter 的 custom_op 参数
torch.onnx.export(
    model,
    dummy_input,
    "model.onnx",
    opset_version=13,
    # 注册 custom op 的 symbolic function
    custom_opsets={}
)
```

### 策略3: 使用 ONNX GraphSurgeon 重写算子

```python
import onnx
import onnx_graphsurgeon as gs
import numpy as np

def replace_einsum_with_matmul(model_path, output_path):
    """将 Einsum 替换为 MatMul 链"""
    graph = gs.import_onnx(onnx.load(model_path))

    nodes_to_remove = []
    nodes_to_add = []

    for node in graph.nodes:
        if node.op == 'Einsum':
            print(f"找到 Einsum 节点: {node.name}")

            # Einsum 的 equation 属性
            equation = node.attrs['equation'].val
            # 例如 'bij,bjk->bik'

            # 分析 equation 并决定如何替换
            # 这里以简化的矩阵乘法为例
            if equation == 'bij,bjk->bik':
                # 创建 MatMul 节点
                inputs = node.inputs
                matmul = gs.Node(
                    op="MatMul",
                    name=f"{node.name}_matmul",
                    inputs=[inputs[0], inputs[1]],
                    outputs=[gs.Variable(f"{node.name}_output", dtype=inputs[0].dtype)]
                )
                nodes_to_add.append(matmul)

            # 标记删除原节点
            nodes_to_remove.append(node)
            # 重连输出
            node.outputs[0].outputs.clear()

    # 应用更改
    graph.nodes.extend(nodes_to_add)
    for node in nodes_to_remove:
        graph.nodes.remove(node)

    graph.cleanup()
    onnx.save(gs.export_onnx(graph), output_path)
    print(f"✅ 算子替换完成，保存到 {output_path}")

# 使用
replace_einsum_with_matmul("model.onnx", "model_fixed.onnx")
```

### 策略4: 算子分解（Decomposition）

某些复杂算子可以分解为多个标准算子：

```python
class DecomposedModel(torch.nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, x):
        # 替代 Einsum: bmm
        A = x[:, :, :64]
        B = x[:, :, 64:]
        batch_size = A.shape[0]
        seq_len = A.shape[1]

        # 使用 Reshape + BatchMatMul
        A_reshaped = A.view(batch_size * seq_len, -1, A.shape[-1])
        B_reshaped = B.view(batch_size * seq_len, B.shape[-2], -1)
        result = torch.bmm(A_reshaped, B_reshaped)
        result = result.view(batch_size, seq_len, -1, A.shape[-1])

        return result

# 或使用 onnx-simplifier 的分解功能
import onnxsim

# 使用 simplify 时指定参数启用分解
model_simplified, check = onnxsim.simplify(
    "model.onnx",
    perform_optimization=True,
    skip_fuse_bn=True,
    # 自定义分解规则（可选）
)
```

## 案例研究：Einsum 算子问题

### 问题
```python
import torch

class AttentionModel(torch.nn.Module):
    def __init__(self, dim=768):
        super().__init__()
        self.scale = dim ** -0.5

    def forward(self, q, k, v):
        # Q: [batch, heads, seq_len, head_dim]
        # K, V: [batch, heads, seq_len, head_dim]
        attn = torch.einsum('bhqd,bhkd->bhqk', q, k) * self.scale
        attn = attn.softmax(dim=-1)
        out = torch.einsum('bhqk,bhkd->bhqd', attn, v)
        return out

model = AttentionModel()
q = torch.randn(2, 12, 128, 64)
k = torch.randn(2, 12, 128, 64)
v = torch.randn(2, 12, 128, 64)

torch.onnx.export(
    model,
    (q, k, v),
    "attention.onnx",
    opset_version=13
)
# 报错: "Einsum is not a symbolic function..."
```

### 解决方案1: 显式替换为 MatMul

```python
class AttentionModelFixed(torch.nn.Module):
    def __init__(self, dim=768):
        super().__init__()
        self.scale = dim ** -0.5

    def forward(self, q, k, v):
        # Q: [batch, heads, seq_len, head_dim]
        # K: [batch, heads, seq_len, head_dim] -> transpose to [batch, heads, head_dim, seq_len]
        k_t = k.transpose(-1, -2)

        # Q * K^T -> [batch, heads, seq_len, seq_len]
        attn = torch.matmul(q, k_t) * self.scale
        attn = attn.softmax(dim=-1)

        # Attention * V -> [batch, heads, seq_len, head_dim]
        out = torch.matmul(attn, v)
        return out

# 重新导出
model = AttentionModelFixed()
torch.onnx.export(
    model,
    (q, k, v),
    "attention_fixed.onnx",
    opset_version=13,
    # 启用符号函数
    custom_opsets=None
)
print("✅ 导出成功，使用 MatMul 替代 Einsum")
```

### 解决方案2: 注册 Einsum 的 symbolic function

```python
# 在自定义 symbolic 文件中定义 einsum
from torch.onnx import symbolic_helper
from torch.onnx.symbolic_helper import _parse_arg

@symbolic_helper.parse_args('v', 'v', 's')
def einsum_symbolic(g, equation, *operands):
    """自定义 Einsum 的 symbolic function"""
    # 解析 equation，例如 "bij,bjk->bik"
    # 支持有限用例，转为 MatMul 和 Reshape
    return g.op("MatMul", operands[0], operands[1])

# 导出时注册
import torch

torch.onnx.export(
    model,
    (q, k, v),
    "attention_custom.onnx",
    opset_version=13,
    custom_opsets={
        'einsum': einsum_symbolic  # 注册自定义 symbolic
    }
)
```

### 解决方案3: 使用 onnxruntime-extensions

```python
# 确保 Einsum 算子存在于 onnxruntime-extensions
from onnxruntime_extensions import get_library_path
import onnxruntime as ort

sess_options = ort.SessionOptions()
sess_options.register_custom_ops_library(get_library_path())

sess = ort.InferenceSession(
    "attention.onnx",
    sess_options=sess_options
)
# Einsum 现在可用
```

## 案例研究：ScatterND 算子问题

### 场景
模型需要根据索引张量更新指定位置的数值（类似 `tensor[indices] = updates`）。

```python
import torch

class ScatterModel(torch.nn.Module):
    def forward(self, base, indices, updates):
        # base: [N, D]
        # indices: [K] - K 个索引
        # updates: [K, D]
        base[indices] = updates  # PyTorch in-place scatter
        return base

model = ScatterModel()
base = torch.randn(100, 512)
indices = torch.randint(0, 100, (10,))
updates = torch.randn(10, 512)

torch.onnx.export(
    model,
    (base, indices, updates),
    "scatter.onnx",
    opset_version=13
)
# 报错: "ScatterND is not supported in this opset version"
```

### 解决方案1: 使用 ScatterElements（如果适用）

```python
class ScatterElementsModel(torch.nn.Module):
    def forward(self, base, indices, updates):
        # ONNX 的 ScatterElements 要求 axis 参数
        # 使用维度 0（批次维度）
        axis = 0
        result = base.scatter_(axis, indices, updates)
        return result

model = ScatterElementsModel()
torch.onnx.export(
    model,
    (base, indices, updates),
    "scatter_elements.onnx",
    opset_version=13,
    # 指定 symbolic
    custom_opsets=None
)

# 验证
import onnx
onnx.check_model("scatter_elements.onnx")
```

### 解决方案2: ONNX GraphSurgeon 重写

```python
import onnx_graphsurgeon as gs
import onnx

def replace_scatter_with_scatter_elements(model_path, output_path):
    """将 ScatterND 替换为 ScatterElements（维度 0 专用）"""
    graph = gs.import_onnx(onnx.load(model_path))

    for node in graph.nodes:
        if node.op == 'ScatterND':
            print(f"替换 {node.name}")

            # 提取输入
            base = node.inputs[0]
            indices = node.inputs[1]
            updates = node.inputs[2]

            # ONNX ScatterElements 需要 axis 参数
            axis_attr = gs.Attr("axis", 0)
            reduction_attr = gs.Attr("reduction", "none")

            scatter_elem = gs.Node(
                op="ScatterElements",
                name=node.name + "_se",
                inputs=[base, indices, updates],
                outputs=node.outputs,
                attrs=[axis_attr, reduction_attr]
            )
            graph.nodes.append(scatter_elem)
            graph.nodes.remove(node)

    graph.cleanup()
    onnx.save(gs.export_onnx(graph), output_path)
    print(f"✅ 替换完成: {output_path}")
```

    except Exception as e:
        print(f"❌ 形状推断失败: {e}")

    print("\n" + "=" * 60)
    print("步骤 4: ONNX Runtime 推理测试")
    print("=" * 60)
    try:
        sess = ort.InferenceSession(model_path)
        input_name = sess.get_inputs()[0].name

        # 确保输入数据类型正确
        expected_type = sess.get_inputs()[0].type

    except Exception as e:
        error_msg = str(e)

        # 解析错误信息
        if "Operator" in error_msg and "not found" in error_msg:
            # 提取算子名
            import re
            match = re.search(r"Operator '([^']+)'", error_msg)
            if match:
                unsupported_op = match.group(1)
                print(f"❌ 不支持的算子: {unsupported_op}")
                print(f"\n解决建议:")
                print(f"  1. 检查 opset 版本（建议 ≥13）")
                print(f"  2. 使用 onnxruntime-extensions: pip install onnxruntime-extensions")
                print(f"  3. 重写模型避免 '{unsupported_op}'")
                print(f"  4. 提供自定义 symbolic function")

        elif "UNSUPPORTED_NODE" in error_msg:
            match = re.search(r"UNSUPPORTED_NODE.*'([^']+)'", error_msg)
            if match:
                unsupported_node = match.group(1)
                print(f"❌ 不支持的节点: {unsupported_node}")

        else:
            print(f"❌ 其他错误: {e}")

## 自定义 Kernel 开发

对于必须使用的算子，可以开发自定义 Kernel：

```cpp
// op_square.cc - 示例：实现 Square 算子
#include <onnxruntime_cxx_api.h>
#include <cmath>

 Ort::Status SquareKernel(const Ort::KernelContext* ctx) {
     const auto* input = ctx->GetInput(0);
     auto* output = ctx->GetOutput(0);

     const float* input_data = input->GetTensorData<float>();
     float* output_data = output->GetTensorData<float>();

     // 计算
     const size_t size = input->GetTensorTypeAndShape().GetElementsCount();
     for (size_t i = 0; i < size; ++i) {
         output_data[i] = input_data[i] * input_data[i];
     }

     return Ort::Status::OK();
 }

 // 注册
 Ort::CustomOpDomain domain("square_domain");
 domain.Add(SquareOp());
 Ort::Env env;
 Ort::SessionOptions session_options;
 session_options.RegisterCustomOpsLibrary("libsquare.so", &domain);
```

## 算子兼容性处理的高级技巧

### 使用函数替换

```python
import torch.nn.functional as F

class CompatibleAttention(torch.nn.Module):
    def forward(self, q, k, v):
        # 避免使用 Einsum，使用标准 API
        # torch.einsum('bhqd,bhkd->bhqk', q, k)
        # -> 使用 matmul
        q = q / (q.shape[-1] ** 0.5)
        scores = torch.matmul(q, k.transpose(-1, -2))  # [batch, heads, seq_len, seq_len]
        attn = F.softmax(scores, dim=-1)
        out = torch.matmul(attn, v)
        return out
```

## 实用工具函数

```python
def check_onnxruntime_op_support(model_path):
    """详细检查算子支持情况"""
    import onnx
    import onnxruntime as ort

    model = onnx.load(model_path)
    ops_in_model = set(node.op_type for node in model.graph.node)

    # ONNX Runtime 核心支持（CPU）
    supported_core = {
        'Add', 'Sub', 'Mul', 'Div', 'MatMul', 'Gemm',
        'Conv', 'MaxPool', 'AveragePool', 'GlobalAveragePool',
        'Relu', 'Sigmoid', 'Tanh', 'Softmax',
        'BatchNormalization', 'Dropout',
        'Reshape', 'Flatten', 'Squeeze', 'Unsqueeze',
        'Concat', 'Slice', 'Split',
        'ReduceMean', 'ReduceSum', 'ReduceMax',
        'Transpose', 'Gather', 'GatherElements',
        'Cast', 'Constant',
        # 更多...
    }

    unsupported_ops = ops_in_model - supported_core

    if unsupported_ops:
        print("⚠️ 以下算子可能需要扩展:")
        for op in unsupported_ops:
            print(f"  - {op}")
    else:
        print("✅ 所有算子均被 ONNX Runtime 核心支持")

    return unsupported_ops
```
