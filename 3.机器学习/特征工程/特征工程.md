# 特征工程

## 1. 引言

特征工程是机器学习项目中最关键、最耗时且最具创造性的环节之一。它涉及从原始数据中提取、转换和选择特征，以构建能够有效训练机器学习模型的数据表示。良好的特征工程可以显著提升模型性能，减少过拟合，提高模型泛化能力，甚至在某些情况下比模型选择更重要。

### 1.1 特征工程的重要性
- **性能提升**：好的特征可以使简单模型达到复杂模型的性能
- **计算效率**：减少特征维度可以降低计算成本
- **可解释性**：精心设计的特征可以提高模型的可解释性
- **数据理解**：特征工程过程帮助深入理解数据特性

### 1.2 特征工程的核心目标
1. **提高模型性能**：通过特征优化提升预测准确率
2. **降低计算复杂度**：减少特征维度，提高训练效率
3. **增强模型鲁棒性**：减少噪声和异常值的影响
4. **提高可解释性**：使模型决策过程更透明

## 2. 特征理解与探索

### 2.1 特征类型

#### 2.1.1 数值型特征
- **连续型特征**：可以在一定范围内取任意值（如温度、价格）
- **离散型特征**：只能取特定值（如年龄、数量）

#### 2.1.2 类别型特征
- **名义特征**：无顺序关系（如颜色、城市）
- **有序特征**：有顺序关系（如评级、教育程度）

#### 2.1.3 文本特征
- **短文本**：标题、标签
- **长文本**：文章、评论
- **结构化文本**：JSON、XML格式的文本

#### 2.1.4 时间序列特征
- **时间戳**：具体时间点
- **时间间隔**：时间段
- **周期性特征**：季节、星期、小时

#### 2.1.5 图像特征
- **像素特征**：原始像素值
- **形状特征**：边缘、轮廓
- **纹理特征**：颜色分布、模式

```python
# 异常值检测方法
def detect_outliers_iqr(data, column):
    """使用IQR方法检测异常值"""
    Q1 = data[column].quantile(0.25)
    Q3 = data[column].quantile(0.75)
    IQR = Q3 - Q1
    lower_bound = Q1 - 1.5 * IQR
    upper_bound = Q3 + 1.5 * IQR
    
    outliers = data[(data[column] < lower_bound) | (data[column] > upper_bound)]
    return outliers, lower_bound, upper_bound

def detect_outliers_zscore(data, column, threshold=3):
    """使用Z-score方法检测异常值"""
    from scipy import stats
    z_scores = np.abs(stats.zscore(data[column].dropna()))
    outliers = data[z_scores > threshold]
    return outliers

# 检测每个数值特征的异常值
```

```python
# 数值特征基础处理
def create_basic_numeric_features(df, numeric_cols):
    """创建基础数值特征"""
    df_features = df.copy()
    
    # 1. 统计特征
    for col in numeric_cols:
        # 分位数特征
        df_features[f'{col}_q25'] = df[col].quantile(0.25)
        df_features[f'{col}_q50'] = df[col].quantile(0.50)
        df_features[f'{col}_q75'] = df[col].quantile(0.75)
        
        # 标准化特征
        df_features[f'{col}_zscore'] = (df[col] - df[col].mean()) / df[col].std()
        
        # 分箱特征
        df_features[f'{col}_bin'] = pd.qcut(df[col], q=5, labels=False, duplicates='drop')
    
    # 2. 交互特征
    for i in range(len(numeric_cols)):
        for j in range(i+1, len(numeric_cols)):
            col1, col2 = numeric_cols[i], numeric_cols[j]
            df_features[f'{col1}_div_{col2}'] = df[col1] / (df[col2] + 1e-10)  # 避免除零
            df_features[f'{col1}_mul_{col2}'] = df[col1] * df[col2]
            df_features[f'{col1}_add_{col2}'] = df[col1] + df[col2]
            df_features[f'{col1}_sub_{col2}'] = df[col1] - df[col2]
    
    return df_features

# 3. 多项式特征
from sklearn.preprocessing import PolynomialFeatures

def create_polynomial_features(df, numeric_cols, degree=2):
    """创建多项式特征"""
    poly = PolynomialFeatures(degree=degree, include_bias=False)
    poly_features = poly.fit_transform(df[numeric_cols])
    
    # 获取特征名称
    feature_names = poly.get_feature_names_out(numeric_cols)
    poly_df = pd.DataFrame(poly_features, columns=feature_names, index=df.index)
    
    return pd.concat([df, poly_df], axis=1)
```

```python
# 高级数值特征构建
def create_advanced_numeric_features(df, numeric_cols):
    """创建高级数值特征"""
    df_features = df.copy()
    
    # 1. 对数变换（处理偏态分布）
    for col in numeric_cols:
        if df[col].min() > 0:  # 确保所有值大于0
            df_features[f'{col}_log'] = np.log1p(df[col])  # log(1+x)
            df_features[f'{col}_log10'] = np.log10(df[col] + 1)
    
    # 2. 指数变换
    for col in numeric_cols:
        df_features[f'{col}_exp'] = np.exp(df[col] / df[col].std())  # 标准化后取指数
    
    # 3. 幂变换（Box-Cox变换）
    from scipy import stats
    
    for col in numeric_cols:
        if df[col].min() > 0:  # Box-Cox要求正值
            transformed, _ = stats.boxcox(df[col] + 1)  # +1避免零值
            df_features[f'{col}_boxcox'] = transformed
    
    # 4. 分位数变换
    from sklearn.preprocessing import QuantileTransformer
    
    qt = QuantileTransformer(output_distribution='normal', random_state=42)
    qt_features = qt.fit_transform(df[numeric_cols])
    qt_df = pd.DataFrame(qt_features, 
                        columns=[f'{col}_quantile' for col in numeric_cols],
                        index=df.index)
    
    return pd.concat([df_features, qt_df], axis=1)
```

```python
# 类别特征编码
from sklearn.preprocessing import LabelEncoder, OneHotEncoder, OrdinalEncoder

def encode_categorical_features(df, categorical_cols):
    """编码类别特征"""
    df_encoded = df.copy()
    
    # 1. Label Encoding（标签编码）
    label_encoders = {}
    for col in categorical_cols:
        if df[col].nunique() <= 10:  # 类别数较少时使用
            le = LabelEncoder()
            df_encoded[f'{col}_label'] = le.fit_transform(df[col].fillna('missing'))
            label_encoders[col] = le
    
    # 2. One-Hot Encoding（独热编码）
    for col in categorical_cols:
        if df[col].nunique() <= 5:  # 类别数很少时使用
            dummies = pd.get_dummies(df[col], prefix=col, dummy_na=True)
            df_encoded = pd.concat([df_encoded, dummies], axis=1)
    
    # 3. Ordinal Encoding（序数编码）
    ordinal_cols = [col for col in categorical_cols if '等级' in col or '级别' in col]
    for col in ordinal_cols:
        # 定义顺序（根据业务逻辑）
        if '等级' in col:
            categories = ['低', '中', '高']
        elif '级别' in col:
            categories = ['一级', '二级', '三级', '四级']
        else:
            categories = sorted(df[col].unique())
        
        oe = OrdinalEncoder(categories=[categories])
        df_encoded[f'{col}_ordinal'] = oe.fit_transform(df[[col]])
    
    return df_encoded
```

#### 3.2.2 高级编码方法
```python
# 高级类别编码技术
def create_advanced_categorical_features(df, categorical_cols, target_col=None):
    """创建高级类别特征"""
    df_features = df.copy()
    
    # 1. 频率编码（Frequency Encoding）
    for col in categorical_cols:
        freq = df[col].value_counts(normalize=True)
        df_features[f'{col}_freq'] = df[col].map(freq)
    
    # 2. 目标编码（Target Encoding）
    if target_col is not None:
        for col in categorical_cols:
            # 计算每个类别的目标均值
            target_mean = df.groupby(col)[target_col].mean()
            df_features[f'{col}_target_mean'] = df[col].map(target_mean)
            
            # 添加平滑处理（避免过拟合）
            global_mean = df[target_col].mean()
            n = df.groupby(col).size()
            m = 10  # 平滑参数
            
            smoothed_means = (target_mean * n + global_mean * m) / (n + m)
            df_features[f'{col}_target_smoothed'] = df[col].map(smoothed_means)
    
    # 3. 证据权重（Weight of Evidence, WoE）
    if target_col is not None:
        for col in categorical_cols:
            # 计算每个类别的正负样本比例
            positive = df[df[target_col] == 1].groupby(col).size()
            negative = df[df[target_col] == 0].groupby(col).size()
            
            # 避免除零错误
            positive = positive.replace(0, 0.5)
            negative = negative.replace(0, 0.5)
            
            # 计算WoE
            woe = np.log(positive / negative)
            df_features[f'{col}_woe'] = df[col].map(woe)
    
    # 4. 类别嵌入（Category Embedding）
    # 使用神经网络学习类别特征的嵌入表示
    # 适用于高基数类别特征
    
    return df_features
```

```python
# 高基数类别特征处理
def handle_high_cardinality_categorical(df, categorical_cols, threshold=20):
    """处理高基数类别特征"""
    df_processed = df.copy()
    
    for col in categorical_cols:
        unique_count = df[col].nunique()
        
        if unique_count > threshold:
            print(f"处理高基数特征 {col}: {unique_count}个唯一值")
            
            # 1. 稀有类别合并
            value_counts = df[col].value_counts()
            rare_categories = value_counts[value_counts < threshold].index
            
            # 将稀有类别合并为"其他"
            df_processed[col] = df[col].apply(
                lambda x: '其他' if x in rare_categories else x
            )
            
            # 2. 基于业务逻辑的分组
            # 根据业务知识将相关类别分组
            
            # 3. 使用哈希技巧（Hashing Trick）
            from sklearn.feature_extraction import FeatureHasher
            
            hasher = FeatureHasher(n_features=10, input_type='string')
            hashed_features = hasher.transform(df[[col]].astype(str).values)
            hashed_df = pd.DataFrame(
                hashed_features.toarray(),
                columns=[f'{col}_hash_{i}' for i in range(10)],
                index=df.index
            )
            
            df_processed = pd.concat([df_processed, hashed_df], axis=1)
    
    return df_processed
```

```python
# 文本预处理
import re
import nltk
from nltk.corpus import stopwords
from nltk.tokenize import word_tokenize
from nltk.stem import PorterStemmer, WordNetLemmatizer

# 下载必要的NLTK数据
nltk.download('punkt', quiet=True)
nltk.download('stopwords', quiet=True)
nltk.download('wordnet', quiet=True)

def preprocess_text(text):
    """文本预处理函数"""
    if not isinstance(text, str):
        return ""
    
    # 1. 转换为小写
    text = text.lower()
    
    # 2. 移除特殊字符和数字
    text = re.sub(r'[^a-zA-Z\s]', '', text)
    
    # 3. 分词
    tokens = word_tokenize(text)
    
    # 4. 移除停用词
    stop_words = set(stopwords.words('english'))
    tokens = [word for word in tokens if word not in stop_words]
    
    # 5. 词干提取
    stemmer = PorterStemmer()
    tokens = [stemmer.stem(word) for word in tokens]
    
    # 6. 词形还原
    lemmatizer = WordNetLemmatizer()
    tokens = [lemmatizer.lemmatize(word) for word in tokens]
    
    # 7. 移除短词
    tokens = [word for word in tokens if len(word) > 2]
    
    return ' '.join(tokens)

def create_text_features(df, text_col):
    """创建文本特征"""
    df_features = df.copy()
    
    # 应用文本预处理
    df_features[f'{text_col}_processed'] = df[text_col].apply(preprocess_text)
    
    # 基础文本特征
    df_features[f'{text_col}_length'] = df[text_col].str.len()
    df_features[f'{text_col}_word_count'] = df[text_col].str.split().str.len()
    df_features[f'{text_col}_char_count'] = df[text_col].str.replace(' ', '').str.len()
    df_features[f'{text_col}_avg_word_length'] = df_features[f'{text_col}_char_count'] / df_features[f'{text_col}_word_count']
    
    # 特殊字符统计
    df_features[f'{text_col}_digit_count'] = df[text_col].str.count(r'\d')
    df_features[f'{text_col}_uppercase_count'] = df[text_col].str.count(r'[A-Z]')
    df_features[f'{text_col}_punctuation_count'] = df[text_col].str.count(r'[^\w\s]')
    
    return df_features
```

```python
# 文本向量化方法
from sklearn.feature_extraction.text import CountVectorizer, TfidfVectorizer
from gensim.models import Word2Vec, FastText

def vectorize_text_features(df, text_col, method='tfidf'):
    """文本向量化"""
    
    if method == 'bow':
        # Bag of Words
        vectorizer = CountVectorizer(
            max_features=1000,
            ngram_range=(1, 2),  # 包含unigram和bigram
            stop_words='english'
        )
        bow_features = vectorizer.fit_transform(df[text_col])
        feature_names = vectorizer.get_feature_names_out()
        
    elif method == 'tfidf':
        # TF-IDF
        vectorizer = TfidfVectorizer(
            max_features=1000,
            ngram_range=(1, 2),
            stop_words='english',
            sublinear_tf=True  # 使用1+log(tf)
        )
        tfidf_features = vectorizer.fit_transform(df[text_col])
        feature_names = vectorizer.get_feature_names_out()
        
    elif method == 'word2vec':
        # Word2Vec
        # 准备训练数据
        sentences = [text.split() for text in df[text_col]]
        
        # 训练Word2Vec模型
        model = Word2Vec(
            sentences,
            vector_size=100,
            window=5,
            min_count=2,
            workers=4,
            sg=1  # skip-gram模型
        )
        
        # 获取文档向量（词向量的平均值）
        def get_doc_vector(text):
            words = text.split()
            word_vectors = [model.wv[word] for word in words if word in model.wv]
            if len(word_vectors) > 0:
                return np.mean(word_vectors, axis=0)
            else:
                return np.zeros(model.vector_size)
        
        w2v_features = np.array([get_doc_vector(text) for text in df[text_col]])
        feature_names = [f'{text_col}_w2v_{i}' for i in range(model.vector_size)]
        
    elif method == 'fasttext':
        # FastText
        sentences = [text.split() for text in df[text_col]]
        
        model = FastText(
            sentences,
            vector_size=100,
            window=5,
            min_count=2,
            workers=4,
            sg=1
        )
        
        def get_doc_vector(text):
            words = text.split()
            word_vectors = [model.wv[word] for word in words if word in model.wv]
            if len(word_vectors) > 0:
                return np.mean(word_vectors, axis=0)
            else:
                return np.zeros(model.vector_size)
        
        ft_features = np.array([get_doc_vector(text) for text in df[text_col]])
        feature_names = [f'{text_col}_ft_{i}' for i in range(model.vector_size)]
    
    # 创建特征DataFrame
    if method in ['bow', 'tfidf']:
        features_df = pd.DataFrame(
            bow_features.toarray() if method == 'bow' else tfidf_features.toarray(),
            columns=feature_names,
            index=df.index
        )
    else:
        features_df = pd.DataFrame(
            w2v_features if method == 'word2vec' else ft_features,
            columns=feature_names,
            index=df.index
        )
    
    return features_df
```

```python
# 使用预训练语言模型
from transformers import AutoTokenizer, AutoModel
import torch

def extract_bert_features(df, text_col, model_name='bert-base-uncased'):
    """使用BERT提取文本特征"""
    
    # 加载预训练模型和分词器
    tokenizer = AutoTokenizer.from_pretrained(model_name)
    model = AutoModel.from_pretrained(model_name)
    
    # 设置为评估模式
    model.eval()
    
    # 提取特征
    bert_features = []
    
    for text in df[text_col]:
        # 分词
        inputs = tokenizer(
            text,
            return_tensors='pt',
            truncation=True,
            padding=True,
            max_length=512
        )
        
        # 前向传播
        with torch.no_grad():
            outputs = model(**inputs)
        
        # 获取[CLS] token的表示作为文档表示
        cls_embedding = outputs.last_hidden_state[:, 0, :].numpy()
        bert_features.append(cls_embedding[0])
    
    # 创建特征DataFrame
    feature_names = [f'{text_col}_bert_{i}' for i in range(len(bert_features[0]))]
    bert_df = pd.DataFrame(bert_features, columns=feature_names, index=df.index)
    
    return bert_df
```

```python
# 时间序列特征提取
def extract_time_features(df, date_col):
    """提取时间特征"""
    df_features = df.copy()
    
    # 确保日期列是datetime类型
    df_features[date_col] = pd.to_datetime(df[date_col])
    
    # 基础时间特征
    df_features[f'{date_col}_year'] = df_features[date_col].dt.year
    df_features[f'{date_col}_month'] = df_features[date_col].dt.month
    df_features[f'{date_col}_day'] = df_features[date_col].dt.day
    df_features[f'{date_col}_hour'] = df_features[date_col].dt.hour
    df_features[f'{date_col}_minute'] = df_features[date_col].dt.minute

    df_features[f'{date_col}_weekofyear'] = df_features[date_col].dt.isocalendar().week

    df_features[f'{date_col}_quarter'] = df_features[date_col].dt.quarter

    if len(df) > 1:
        df_features[f'{date_col}_time_since_last'] = df_features[date_col].diff().dt.total_seconds()
        df_features[f'{date_col}_time_to_next'] = df_features[date_col].diff(-1).dt.total_seconds().abs()
    
    # 周期性特征（使用正弦余弦编码）
    df_features[f'{date_col}_month_sin'] = np.sin(2 * np.pi * df_features[f'{date_col}_month'] / 12)
```

```python
# 传统图像特征提取
import cv2
from skimage import feature, filters, measure

def extract_traditional_image_features(image_path):
    """提取传统图像特征"""
    # 读取图像
    image = cv2.imread(image_path)
    if image is None:
        return {}
    
    # 转换为灰度图
    gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
    
    features = {}
    
    # 1. 颜色特征
    # 颜色直方图
    color_hist = cv2.calcHist([image], [0, 1, 2], None, [8, 8, 8], [0, 256, 0, 256, 0, 256])
    color_hist = color_hist.flatten()
    for i, val in enumerate(color_hist[:10]):  # 取前10个bin
        features[f'color_hist_{i}'] = val
    
    # 颜色矩（均值、标准差、偏度）
    for channel in range(3):
        channel_data = image[:, :, channel].flatten()
        features[f'color_mean_ch{channel}'] = np.mean(channel_data)
        features[f'color_std_ch{channel}'] = np.std(channel_data)
        features[f'color_skew_ch{channel}'] = stats.skew(channel_data)
    
    # 2. 纹理特征
    # GLCM（灰度共生矩阵）特征
    glcm = feature.graycomatrix(gray, distances=[1], angles=[0], symmetric=True, normed=True)
    glcm_features = feature.graycoprops(glcm, ['contrast', 'dissimilarity', 'homogeneity', 'energy', 'correlation', 'ASM'])
    
    for prop in ['contrast', 'dissimilarity', 'homogeneity', 'energy', 'correlation', 'ASM']:
        features[f'glcm_{prop}'] = glcm_features[prop][0, 0]
    
    # LBP（局部二值模式）
    lbp = feature.local_binary_pattern(gray, P=8, R=1, method='uniform')
    lbp_hist, _ = np.histogram(lbp.ravel(), bins=np.arange(0, 11), range=(0, 10))
    lbp_hist = lbp_hist.astype('float') / lbp_hist.sum()
    
    for i, val in enumerate(lbp_hist):
        features[f'lbp_bin_{i}'] = val
    
    # 3. 形状特征
    # 边缘检测
    edges = filters.sobel(gray)
    features['edge_density'] = np.mean(edges > 0.1)
    
    # 轮廓特征
    _, binary = cv2.threshold(gray, 127, 255, cv2.THRESH_BINARY)
    contours, _ = cv2.findContours(binary, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
    
    if len(contours) > 0:
        largest_contour = max(contours, key=cv2.contourArea)
        features['contour_area'] = cv2.contourArea(largest_contour)
        features['contour_perimeter'] = cv2.arcLength(largest_contour, True)
        
        # 形状描述符
        hull = cv2.convexHull(largest_contour)
        features['convexity'] = cv2.contourArea(largest_contour) / cv2.contourArea(hull)
    
    # 4. 统计特征
    features['image_mean'] = np.mean(gray)
    features['image_std'] = np.std(gray)
    features['image_skew'] = stats.skew(gray.flatten())
    features['image_kurtosis'] = stats.kurtosis(gray.flatten())
    
    return features
```

```python
def extract_cnn_features(image_path, model_name='resnet50'):
    """使用预训练CNN提取图像特征"""
    
    # 加载预训练模型
    if model_name == 'resnet50':
        model = models.resnet50(pretrained=True)
    elif model_name == 'vgg16':
        model = models.vgg16(pretrained=True)
    elif model_name == 'efficientnet':
        model = models.efficientnet_b0(pretrained=True)
    else:
        raise ValueError(f"不支持的模型: {model_name}")
    
    # 移除最后的全连接层
    model = torch.nn.Sequential(*list(model.children())[:-1])
    model.eval()
    
    # 图像预处理
    transform = transforms.Compose([
        transforms.Resize(256),
        transforms.CenterCrop(224),
        transforms.ToTensor(),
        transforms.Normalize(mean=[0.485, 0.456, 0.406], std=[0.229, 0.224, 0.225])
    ])
    
    # 加载图像
    image = Image.open(image_path).convert('RGB')
    image_tensor = transform(image).unsqueeze(0)
    
    # 提取特征
    with torch.no_grad():
        features = model(image_tensor)
    
    # 展平特征
    features = features.squeeze().numpy()
    
    return features

def extract_multiple_cnn_features(image_paths, model_names=['resnet50', 'vgg16']):
    """提取多个CNN模型的特征"""
    all_features = {}
```

```python
        # 方差分析（ANOVA）F值，适用于分类问题

    elif method == 'f_regression':

        selector = SelectKBest(score_func=chi2, k=k)
    elif method == 'mutual_info_classif':
        # 互信息，适用于分类问题
        selector = SelectKBest(score_func=mutual_info_classif, k=k)
    elif method == 'mutual_info_regression':
        # 互信息，适用于回归问题
        selector = SelectKBest(score_func=mutual_info_regression, k=k)
    else:
        raise ValueError(f"不支持的过滤方法: {method}")
    
    # 拟合选择器

    selected_features = selector.get_support()
```

```python
# 基于相关性的特征选择
def correlation_based_feature_selection(X, y, threshold=0.8):
    """基于相关性的特征选择"""
    
    # 1. 计算特征与目标变量的相关性
    target_correlation = X.apply(lambda col: col.corr(y))
    
    # 2. 计算特征间的相关性
    feature_correlation = X.corr().abs()
    
    # 3. 选择与目标变量相关性高的特征
    high_target_corr_features = target_correlation[abs(target_correlation) > 0.1].index.tolist()
    
    # 4. 移除高度相关的特征
    selected_features = []
    dropped_features = []
    
    for i, feature_i in enumerate(high_target_corr_features):
        if feature_i in dropped_features:
            continue
            
        selected_features.append(feature_i)
        
        # 检查与其他特征的相关性
        for j, feature_j in enumerate(high_target_corr_features[i+1:]):
            if feature_j in dropped_features:
                continue
                
            # 如果两个特征高度相关，保留与目标变量相关性更高的那个
            if feature_correlation.loc[feature_i, feature_j] > threshold:
                if abs(target_correlation[feature_i]) >= abs(target_correlation[feature_j]):
                    dropped_features.append(feature_j)
                else:
                    # 如果当前特征相关性更低，移除它并跳出循环
                    selected_features.remove(feature_i)
                    dropped_features.append(feature_i)
                    break
    
    # 5. 返回选择的特征
    X_selected = X[selected_features]
    
    return X_selected, selected_features, dropped_features

# 使用示例
X_corr_selected, selected, dropped = correlation_based_feature_selection(X_train, y_train, threshold=0.85)
print(f"选择了 {len(selected)} 个特征")
print(f"移除了 {len(dropped)} 个高度相关的特征")
```

```python
from sklearn.linear_model import Lasso, Ridge, ElasticNet
from sklearn.svm import LinearSVC, LinearSVR

def regularization_based_feature_selection(X, y, model_type='lasso', alpha=0.01):
    """基于正则化的特征选择"""
    
    if model_type == 'lasso':
        # L1正则化（Lasso），会产生稀疏解
        model = Lasso(alpha=alpha, random_state=42)
    elif model_type == 'ridge':
        # L2正则化（Ridge），不会产生稀疏解
        model = Ridge(alpha=alpha, random_state=42)
    elif model_type == 'elasticnet':
        # ElasticNet，L1和L2的组合
        model = ElasticNet(alpha=alpha, l1_ratio=0.5, random_state=42)
    elif model_type == 'linearsvc':
        # 线性SVM分类器，使用L2正则化
        model = LinearSVC(C=1/alpha, penalty='l2', dual=False, random_state=42)
    elif model_type == 'linearsvr':
        # 线性SVM回归器，使用L2正则化
        model = LinearSVR(C=1/alpha, random_state=42)
    else:
        raise ValueError(f"不支持的模型类型: {model_type}")
    
    # 训练模型
    model.fit(X, y)
    
    # 获取特征重要性（系数绝对值）
    if hasattr(model, 'coef_'):
        feature_importance = np.abs(model.coef_).flatten()
    else:
        # 对于某些模型，可能需要使用feature_importances_
        feature_importance = model.feature_importances_
    
    # 创建特征重要性DataFrame
    importance_df = pd.DataFrame({
        'feature': X.columns,
        'importance': feature_importance,
        'selected': feature_importance > 0
    }).sort_values('importance', ascending=False)
    
    # 选择非零系数的特征
    selected_features = importance_df[importance_df['selected']]['feature'].tolist()
    X_selected = X[selected_features]
    
    return X_selected, importance_df
```

### 4.4 特征选择策略比较与选择

#### 4.4.1 特征选择方法比较
| 方法类型 | 优点 | 缺点 | 适用场景 |
|---------|------|------|---------|
| **过滤式** | 计算快，独立于模型，可扩展性好 | 忽略特征交互，可能选择冗余特征 | 大规模数据，初步特征筛选 |
| **包裹式** | 考虑特征交互，通常性能更好 | 计算成本高，可能过拟合 | 中小规模数据，追求最佳性能 |
| **嵌入式** | 结合了过滤式和包裹式的优点 | 依赖于特定模型 | 模型训练过程中自动选择 |

#### 4.4.2 特征选择流程建议
1. **初步筛选**：使用过滤式方法（如方差阈值、相关性）移除明显无用的特征
2. **模型相关选择**：使用嵌入式方法（如L1正则化、树模型）进行特征选择
3. **精细调整**：使用包裹式方法（如RFE、顺序选择）优化特征子集
4. **验证评估**：使用交叉验证评估不同特征子集的性能

#### 4.4.3 特征选择最佳实践

```python
def encode_numeric_features(X, method='binning', **kwargs):
    """数值特征编码"""
    
    if method == 'binning':
        # 分箱编码
        n_bins = kwargs.get('n_bins', 5)
        strategy = kwargs.get('strategy', 'uniform')  # 'uniform', 'quantile', 'kmeans'
        
        from sklearn.preprocessing import KBinsDiscretizer
        
        encoder = KBinsDiscretizer(
            n_bins=n_bins,
            encode='ordinal',  # 'ordinal', 'onehot', 'onehot-dense'
            strategy=strategy
        )
        
    elif method == 'polynomial':
        # 多项式编码
        degree = kwargs.get('degree', 2)
        interaction_only = kwargs.get('interaction_only', False)
        
        from sklearn.preprocessing import PolynomialFeatures
        
        encoder = PolynomialFeatures(
            degree=degree,
            interaction_only=interaction_only,
            include_bias=False
        )
        
    elif method == 'spline':
        # 样条基函数编码
        n_knots = kwargs.get('n_knots', 5)

        from sklearn.preprocessing import SplineTransformer
        
        encoder = SplineTransformer(
            n_knots=n_knots,
            degree=degree,
            include_bias=False
        )
        
    else:
        raise ValueError(f"不支持的数值编码方法: {method}")
    
    # 拟合并转换数据
    X_encoded = encoder.fit_transform(X)
    
    return X_encoded, encoder
```

```python
from sklearn.preprocessing import OneHotEncoder, OrdinalEncoder, TargetEncoder
from category_encoders import BinaryEncoder, CountEncoder, WOEEncoder

def encode_categorical_features(X, y=None, method='onehot', **kwargs):
    """类别特征编码"""
    
    if method == 'onehot':
        # 独热编码
        handle_unknown = kwargs.get('handle_unknown', 'ignore')
        sparse_output = kwargs.get('sparse_output', False)
        
        encoder = OneHotEncoder(
            handle_unknown=handle_unknown,
            sparse_output=sparse_output
        )
        
    elif method == 'ordinal':
        # 序数编码
        categories = kwargs.get('categories', 'auto')
        unknown_value = kwargs.get('unknown_value', None)
        
        encoder = OrdinalEncoder(
            categories=categories,
            unknown_value=unknown_value
        )
        
    elif method == 'target':
        # 目标编码（需要目标变量y）
        if y is None:
            raise ValueError("目标编码需要目标变量y")
        
        smooth = kwargs.get('smooth', 'auto')
        
        encoder = TargetEncoder(
            smooth=smooth,
            random_state=42
        )
        
    elif method == 'binary':
        # 二进制编码
        encoder = BinaryEncoder()
        
    elif method == 'count':
        # 计数编码
        normalize = kwargs.get('normalize', False)
        
        encoder = CountEncoder(normalize=normalize)
        
    elif method == 'woe':
        # 证据权重编码（需要目标变量y）
        if y is None:
            raise ValueError("WOE编码需要目标变量y")
        
        encoder = WOEEncoder()
        
    else:
        raise ValueError(f"不支持的类别编码方法: {method}")
    
    # 拟合并转换数据
    if method in ['target', 'woe']:
        X_encoded = encoder.fit_transform(X, y)
    else:
        X_encoded = encoder.fit_transform(X)
    
    return X_encoded, encoder
```

```python
    elif method == 'hashing':
        # 哈希编码（适用于大规模数据）
        n_features = kwargs.get('n_features', 2**10)
        ngram_range = kwargs.get('ngram_range', (1, 2))
        
        encoder = HashingVectorizer(
            n_features=n_features,
            ngram_range=ngram_range,
            alternate_sign=False
        )
        
    else:
        raise ValueError(f"不支持的文本编码方法: {method}")
    
    # 拟合并转换数据
    X_encoded = encoder.fit_transform(text_series)
    
    return X_encoded, encoder
```

```python
# 线性降维
from sklearn.decomposition import PCA, TruncatedSVD, FactorAnalysis
from sklearn.discriminant_analysis import LinearDiscriminantAnalysis

def linear_dimensionality_reduction(X, y=None, method='pca', **kwargs):
    """线性降维"""
    
    if method == 'pca':
        # 主成分分析（PCA）
        n_components = kwargs.get('n_components', 2)
        whiten = kwargs.get('whiten', False)
        
        reducer = PCA(
            n_components=n_components,
            whiten=whiten,
            random_state=42
        )
        
    elif method == 'svd':
        # 截断奇异值分解（Truncated SVD）
        n_components = kwargs.get('n_components', 2)
        algorithm = kwargs.get('algorithm', 'randomized')
        
        reducer = TruncatedSVD(
            n_components=n_components,
            algorithm=algorithm,
            random_state=42
        )
        
    elif method == 'fa':
        # 因子分析（Factor Analysis）
        n_components = kwargs.get('n_components', 2)
        
        reducer = FactorAnalysis(
            n_components=n_components,
            random_state=42
        )
        
    elif method == 'lda':
        # 线性判别分析（LDA，需要目标变量y）
        if y is None:
            raise ValueError("LDA需要目标变量y")
        
        n_components = kwargs.get('n_components', min(X.shape[1], len(np.unique(y))-1))
        
        reducer = LinearDiscriminantAnalysis(
            n_components=n_components
        )
        
    else:
        raise ValueError(f"不支持的线性降维方法: {method}")
    
    # 拟合并转换数据
    if method == 'lda':
        X_reduced = reducer.fit_transform(X, y)
    else:
        X_reduced = reducer.fit_transform(X)
    
    # 获取解释方差
    if hasattr(reducer, 'explained_variance_ratio_'):
        explained_variance = reducer.explained_variance_ratio_
        cumulative_variance = np.cumsum(explained_variance)
    else:
        explained_variance = None
        cumulative_variance = None
    
    return X_reduced, reducer, explained_variance, cumulative_variance
```

```python
# 非线性降维
from sklearn.manifold import TSNE, Isomap, MDS, SpectralEmbedding
from umap import UMAP

def nonlinear_dimensionality_reduction(X, method='tsne', **kwargs):
    """非线性降维"""
    
    if method == 'tsne':
        # t-SNE（t-distributed Stochastic Neighbor Embedding）
        n_components = kwargs.get('n_components', 2)
        perplexity = kwargs.get('perplexity', 30)
        learning_rate = kwargs.get('learning_rate', 200)
        
        reducer = TSNE(
            n_components=n_components,
            perplexity=perplexity,
            learning_rate=learning_rate,
            random_state=42
        )
        
    elif method == 'umap':
        # UMAP（Uniform Manifold Approximation and Projection）
        n_components = kwargs.get('n_components', 2)
        n_neighbors = kwargs.get('n_neighbors', 15)
        min_dist = kwargs.get('min_dist', 0.1)
        
        reducer = UMAP(
            n_components=n_components,
            n_neighbors=n_neighbors,
            min_dist=min_dist,
            random_state=42
        )
        
    elif method == 'isomap':
        # Isomap（等距映射）
        n_components = kwargs.get('n_components', 2)
        n_neighbors = kwargs.get('n_neighbors', 5)
        
        reducer = Isomap(
            n_components=n_components,
            n_neighbors=n_neighbors
        )
        
    elif method == 'mds':
        # MDS（多维缩放）
        n_components = kwargs.get('n_components', 2)
        metric = kwargs.get('metric', True)  # True for metric MDS, False for non-metric
        
        reducer = MDS(
            n_components=n_components,
            metric=metric,
            random_state=42
        )
        
    elif method == 'spectral':
        # 谱嵌入
        n_components = kwargs.get('n_components', 2)
        
        reducer = SpectralEmbedding(
            n_components=n_components,
            random_state=42
        )
        
    else:
        raise ValueError(f"不支持的非线性降维方法: {method}")
    
    # 拟合并转换数据
    X_reduced = reducer.fit_transform(X)
    
    return X_reduced, reducer
```

```python
# 自动编码器降维
import tensorflow as tf
from tensorflow.keras import layers, models

def autoencoder_dimensionality_reduction(X, encoding_dim=32, epochs=50, batch_size=32):
    """自动编码器降维"""
    
    # 数据标准化
    from sklearn.preprocessing import StandardScaler
    scaler = StandardScaler()
    X_scaled = scaler.fit_transform(X)
    
    # 构建自动编码器
    input_dim = X_scaled.shape[1]
    
    # 编码器
    input_layer = layers.Input(shape=(input_dim,))
    encoded = layers.Dense(encoding_dim * 2, activation='relu')(input_layer)
    encoded = layers.Dense(encoding_dim, activation='relu')(encoded)
    
    # 解码器
    decoded = layers.Dense(encoding_dim * 2, activation='relu')(encoded)
    decoded = layers.Dense(input_dim, activation='linear')(decoded)
    
    # 构建自动编码器模型
    autoencoder = models.Model(input_layer, decoded)
    
    # 构建编码器模型（用于特征提取）
    encoder = models.Model(input_layer, encoded)
    
    # 编译模型
    autoencoder.compile(optimizer='adam', loss='mse')
    
    # 训练自动编码器
    history = autoencoder.fit(
        X_scaled, X_scaled,
        epochs=epochs,
        batch_size=batch_size,
        shuffle=True,
        validation_split=0.2,
        verbose=0
    )
    
    # 使用编码器提取特征
    X_encoded = encoder.predict(X_scaled)
    
    # 计算重构误差
    X_reconstructed = autoencoder.predict(X_scaled)
    reconstruction_error = np.mean(np.square(X_scaled - X_reconstructed))
    
    return X_encoded, encoder, autoencoder, history, reconstruction_error
```

### 5.4 特征转换最佳实践

#### 5.4.1 转换流程建议
1. **数据清洗**：处理缺失值、异常值
2. **特征编码**：将类别特征转换为数值特征
3. **特征缩放**：标准化或归一化数值特征
4. **特征生成**：创建交互特征、多项式特征等
5. **特征选择**：移除不相关或冗余特征
6. **特征降维**：当特征维度很高时进行降维

#### 5.4.2 转换策略选择
- **数值特征**：标准化、归一化、分箱、多项式扩展
- **类别特征**：独热编码、目标编码、频率编码
- **文本特征**：TF-IDF、词向量、预训练语言模型
- **高维数据**：PCA、t-SNE、UMAP、自动编码器

```python
# 避免数据泄露的特征转换流程
from sklearn.pipeline import Pipeline
from sklearn.model_selection import train_test_split

def safe_feature_transformation(X, y, test_size=0.2):
    """安全的特征转换流程（避免数据泄露）"""
    
    # 划分训练集和测试集
    X_train, X_test, y_train, y_test = train_test_split(
        X, y, test_size=test_size, random_state=42
    )
    
    # 创建转换管道
    from sklearn.compose import ColumnTransformer
    from sklearn.impute import SimpleImputer
    
    # 定义数值特征和类别特征
    numeric_features = X.select_dtypes(include=['int64', 'float64']).columns
    categorical_features = X.select_dtypes(include=['object', 'category']).columns
    
    # 创建数值特征转换器
    numeric_transformer = Pipeline(steps=[
        ('imputer', SimpleImputer(strategy='median')),
        ('scaler', StandardScaler())
    ])
    
    # 创建类别特征转换器
    categorical_transformer = Pipeline(steps=[
        ('imputer', SimpleImputer(strategy='constant', fill_value='missing')),
        ('onehot', OneHotEncoder(handle_unknown='ignore', sparse_output=False))
    ])
    
    # 创建列转换器
    preprocessor = ColumnTransformer(
        transformers=[
            ('num', numeric_transformer, numeric_features),
            ('cat', categorical_transformer, categorical_features)
        ]
    )
    
    # 在训练集上拟合转换器
    X_train_transformed = preprocessor.fit_transform(X_train)
    X_test_transformed = preprocessor.transform(X_test)  # 仅转换，不重新拟合
    
    # 获取特征名称
    feature_names = []
    
    # 数值特征名称
    feature_names.extend(numeric_features.tolist())
    
    # 类别特征名称（经过独热编码）
    if len(categorical_features) > 0:
        onehot = preprocessor.named_transformers_['cat'].named_steps['onehot']
        cat_feature_names = onehot.get_feature_names_out(categorical_features)
        feature_names.extend(cat_feature_names)
    
    return X_train_transformed, X_test_transformed, y_train, y_test, preprocessor, feature_names

# 使用示例
X_train_transformed, X_test_transformed, y_train_transformed, y_test_transformed, preprocessor, feature_names = safe_feature_transformation(
    X, y, test_size=0.2
)

print(f"训练集转换后形状: {X_train_transformed.shape}")
print(f"测试集转换后形状: {X_test_transformed.shape}")
print(f"特征数量: {len(feature_names)}")
```

```python
# 客户流失预测特征工程
def customer_churn_feature_engineering(df):
    """客户流失预测特征工程"""
    
    print("=== 客户流失预测特征工程 ===")
    
    # 1. 数据预处理
    print("\n1. 数据预处理...")
    
    # 处理缺失值
    df_processed = df.copy()
    
    # 数值特征用中位数填充
    numeric_cols = df.select_dtypes(include=[np.number]).columns
    for col in numeric_cols:
        if df[col].isnull().any():
            df_processed[col] = df[col].fillna(df[col].median())
    
    # 类别特征用众数填充
    categorical_cols = df.select_dtypes(include=['object']).columns
    for col in categorical_cols:
        if df[col].isnull().any():
            df_processed[col] = df[col].fillna(df[col].mode()[0])
    
    # 2. 特征创建
    print("\n2. 特征创建...")
    
    # 创建客户价值特征
    if 'MonthlyCharges' in df_processed.columns and 'Tenure' in df_processed.columns:
        df_processed['TotalSpent'] = df_processed['MonthlyCharges'] * df_processed['Tenure']
    
    # 创建服务使用特征
    service_columns = [col for col in df_processed.columns if 'Service' in col or 'Streaming' in col]
    if service_columns:
        df_processed['TotalServices'] = df_processed[service_columns].apply(
            lambda row: sum([1 for val in row if val == 'Yes']), axis=1
        )
    
    # 创建客户忠诚度特征
    if 'Tenure' in df_processed.columns:
        df_processed['TenureGroup'] = pd.cut(
            df_processed['Tenure'],
            bins=[0, 12, 24, 36, 48, 60, np.inf],
            labels=['0-1年', '1-2年', '2-3年', '3-4年', '4-5年', '5年以上']
        )
    
    # 3. 特征编码
    print("\n3. 特征编码...")
    
    from sklearn.preprocessing import LabelEncoder
    
    # 标签编码类别特征
    label_encoders = {}
    for col in categorical_cols:
        if df_processed[col].nunique() <= 10:  # 类别数较少时使用标签编码
            le = LabelEncoder()
            df_processed[f'{col}_encoded'] = le.fit_transform(df_processed[col])
            label_encoders[col] = le
    
    # 独热编码其他类别特征
    for col in categorical_cols:
        if col not in label_encoders and df_processed[col].nunique() <= 5:
            dummies = pd.get_dummies(df_processed[col], prefix=col, drop_first=True)
            df_processed = pd.concat([df_processed, dummies], axis=1)
    
    # 4. 特征选择
    print("\n4. 特征选择...")
    
    # 准备特征和目标变量
    feature_cols = [col for col in df_processed.columns 
                   if col not in ['customerID', 'Churn'] and not col.endswith('_No')]
    
    X = df_processed[feature_cols].select_dtypes(include=[np.number])
    y = df_processed['Churn'].map({'Yes': 1, 'No': 0}) if 'Churn' in df_processed.columns else None
    
    if y is not None:
        from sklearn.feature_selection import SelectKBest, chi2
        
        # 使用卡方检验选择特征
        selector = SelectKBest(score_func=chi2, k=min(20, X.shape[1]))
        X_selected = selector.fit_transform(X, y)
        
        selected_features = X.columns[selector.get_support()].tolist()
        print(f"选择了 {len(selected_features)} 个特征")
        
        return X_selected, y, selected_features
    
    return X, None, feature_cols
```

```python
# 特征稳定性评估
def evaluate_feature_stability(X_train, X_test, feature_names):
    """评估特征稳定性"""
    
    stability_metrics = {}
    
    for i, feature in enumerate(feature_names):
        # 计算训练集和测试集的统计差异
        train_mean = X_train[:, i].mean()
        test_mean = X_test[:, i].mean()
        train_std = X_train[:, i].std()
        test_std = X_test[:, i].std()
        
        # 计算均值差异（标准化）
        mean_diff = abs(train_mean - test_mean) / (train_std + 1e-10)
        
        # 计算分布差异（KS检验）
        from scipy import stats
        ks_statistic, ks_pvalue = stats.ks_2samp(X_train[:, i], X_test[:, i])
        
        stability_metrics[feature] = {
            'train_mean': train_mean,
            'test_mean': test_mean,
            'mean_diff': mean_diff,
            'ks_statistic': ks_statistic,
            'ks_pvalue': ks_pvalue,
            'is_stable': mean_diff < 0.1 and ks_pvalue > 0.05  # 稳定性阈值
        }
    
    # 创建稳定性DataFrame
    stability_df = pd.DataFrame(stability_metrics).T
    stability_df = stability_df.sort_values('mean_diff', ascending=False)
    
    return stability_df
```

```python
# 数据漂移检测
def detect_data_drift(X_reference, X_current, feature_names):
    """检测数据漂移"""
    
    drift_results = {}
    
    for i, feature in enumerate(feature_names):
        # 计算PSI（Population Stability Index）
        def calculate_psi(expected, actual, buckets=10):
            """计算PSI"""
            # 分箱
            breakpoints = np.percentile(expected, np.linspace(0, 100, buckets + 1))
            breakpoints = np.unique(breakpoints)
            
            # 计算每个区间的比例
            expected_percents = np.histogram(expected, breakpoints)[0] / len(expected)
            actual_percents = np.histogram(actual, breakpoints)[0] / len(actual)
            
            # 避免除零
            expected_percents = np.where(expected_percents == 0, 0.0001, expected_percents)
            actual_percents = np.where(actual_percents == 0, 0.0001, actual_percents)
            
            # 计算PSI
            psi = np.sum((actual_percents - expected_percents) * 
                        np.log(actual_percents / expected_percents))
            
            return psi
        
        # 计算PSI
        psi = calculate_psi(X_reference[:, i], X_current[:, i])
        
        # 判断是否漂移
        if psi < 0.1:
            drift_status = "无显著漂移"
        elif psi < 0.25:
            drift_status = "中等漂移"
        else:
            drift_status = "显著漂移"
        
        drift_results[feature] = {
            'psi': psi,
            'drift_status': drift_status
        }
    
    # 创建漂移检测结果DataFrame
    drift_df = pd.DataFrame(drift_results).T
    drift_df = drift_df.sort_values('psi', ascending=False)
    
    return drift_df
```

```python
# 使用自动编码器进行特征学习
def autoencoder_feature_learning(X, encoding_dim=32):
    """使用自动编码器学习特征表示"""
    
    from tensorflow.keras.models import Model
    from tensorflow.keras.layers import Input, Dense
    
    # 构建自动编码器
    input_dim = X.shape[1]
    
    # 编码器
    input_layer = Input(shape=(input_dim,))
    encoded = Dense(encoding_dim * 2, activation='relu')(input_layer)
    encoded = Dense(encoding_dim, activation='relu')(encoded)
    
    # 解码器
    decoded = Dense(encoding_dim * 2, activation='relu')(encoded)
    decoded = Dense(input_dim, activation='sigmoid')(decoded)
    
    # 构建模型
    autoencoder = Model(input_layer, decoded)
    encoder = Model(input_layer, encoded)
    
    # 编译模型
    autoencoder.compile(optimizer='adam', loss='mse')
    
    # 训练模型
    autoencoder.fit(X, X, epochs=50, batch_size=32, verbose=0)
    
    # 提取特征
    features = encoder.predict(X)
    
    return features, encoder, autoencoder
```

```python
# 使用预训练模型提取特征
def pretrained_model_feature_extraction(images, model_name='resnet50'):
    """使用预训练CNN模型提取图像特征"""
    
    from tensorflow.keras.applications import ResNet50, VGG16, EfficientNetB0
    from tensorflow.keras.applications.resnet50 import preprocess_input
    from tensorflow.keras.models import Model
    
    # 加载预训练模型
    if model_name == 'resnet50':
        base_model = ResNet50(weights='imagenet', include_top=False)
    elif model_name == 'vgg16':
        base_model = VGG16(weights='imagenet', include_top=False)
    elif model_name == 'efficientnet':
        base_model = EfficientNetB0(weights='imagenet', include_top=False)
    else:
        raise ValueError(f"不支持的模型: {model_name}")
    
    # 创建特征提取模型
    feature_extractor = Model(inputs=base_model.input, 
                             outputs=base_model.output)
    
    # 预处理图像
    images_processed = preprocess_input(images)
    
    # 提取特征
    features = feature_extractor.predict(images_processed)
    
    # 展平特征
    features_flat = features.reshape(features.shape[0], -1)
    
    return features_flat, feature_extractor
```
