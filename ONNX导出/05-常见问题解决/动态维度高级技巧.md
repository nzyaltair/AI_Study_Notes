# 动态维度高级技巧

## 概述

动态维度（Dynamic Axes）是 ONNX 导出中最强大的功能之一，也是常见错误的根源。本指南从基础到高级，深入讲解如何正确处理可变形状的模型。

**标签**: #dynamic-dimensions #advanced-techniques

---

## 动态维度原理

### 什么是动态维度？

传统 ONNX 模型需要**固定的输入形状**（例如 `[1, 3, 224, 224]`），而动态维度允许某些维度在推理时变化：

```python
# 固定维度
input: [batch=1, channels=3, height=224, width=224]

# 动态维度（批处理）
input: [batch=-1, channels=3, height=224, width=224]
# 可以推理 batch=1, 2, 4, 8...

# 动态维度（图像尺寸）
input: [batch=1, channels=3, height=-1, width=-1]
# 可以推理 (1,3,640,640) 或 (1,3,480,480)

# 完全动态
input: [batch=-1, channels=-1, height=-1, width=-1]
# 任何形状都能推理（谨慎使用，可能限制图优化）
```

## 基础用法回顾

### PyTorch 动态轴导出

```python
import torch

class DynamicModel(torch.nn.Module):
    def forward(self, x, lengths):
        return x

model = DynamicModel()

# 动态批次 + 动态序列长度（NLP 场景）
dynamic_axes = {
    'input': {0: 'batch', 1: 'seq_len'},  # 0轴=批次，1轴=序列长度
    'output': {0: 'batch', 1: 'seq_len'}
}

dummy_input = torch.randn(2, 10, 768)  # batch=2, seq_len=10
torch.onnx.export(
    model,
    dummy_input,
    "model_dynamic.onnx",
    dynamic_axes=dynamic_axes,
    opset_version=13,
    input_names=["input"],
    output_names=["output"]
)
print("✅ 导出成功，维度: [batch, seq_len, 768]")

# 推理时使用不同形状
import onnxruntime as ort
import numpy as np

sess = ort.InferenceSession("model_dynamic.onnx")

# 批次=4，序列=20
input_data = np.random.randn(4, 20, 768).astype(np.float32)
result = sess.run(None, {"input": input_data})
print(f"输出形状: {result[0].shape}")  # (4, 20, 768)
```

## 高级技巧

### 技巧1: 多输入/输出的动态维度

```python
import torch

class MultiInputModel(torch.nn.Module):
    def forward(self, images, boxes, scores):
        # images: [B, C, H, W]
        # boxes: [B, N, 4] - B 个图像，每张 N 个边界框
        # scores: [B, N] - 每张图像的 N 个置信度
        return boxes, scores

model = MultiInputModel()

dynamic_axes = {
    # 输入1（图像）：batch 和 H/W 动态
    'images': {0: 'batch', 2: 'height', 3: 'width'},
    # 输入2（边界框）：batch 和 N 动态
    'boxes': {0: 'batch', 1: 'num_boxes'},
    # 输入3（置信度）：batch 和 N 动态
    'scores': {0: 'batch', 1: 'num_boxes'},
    # 输出：与对应输入保持相同维度
    'boxes_out': {0: 'batch', 1: 'num_boxes'},
    'scores_out': {0: 'batch', 1: 'num_boxes'}
}

dummy_inputs = (
    torch.randn(2, 3, 640, 640),      # batch=2, H=640, W=640
    torch.randn(2, 100, 4),            # batch=2, 100个边界框
    torch.randn(2, 100)               # batch=2, 100个置信度
)

torch.onnx.export(
    model,
    dummy_inputs,
    "multi_input_dynamic.onnx",
    dynamic_axes=dynamic_axes,
    opset_version=13,
    input_names=["images", "boxes", "scores"],
    output_names=["boxes_out", "scores_out"]
)

print("✅ 多输入动态模型导出成功")
```

### 技巧2: 嵌套结构的动态维度

某些模型返回字典或元组，需要特殊处理：

```python
import torch

class ComplexModel(torch.nn.Module):
    def forward(self, x):
        # 返回多个输出
        features = self.backbone(x)
        detections = self.head(features)
        return {
            'features': features,
            'detections': detections,
            'scores': detections[:, :, 4]
        }

# 问题：ONNX 导出不支持字典返回

# 解决方案1: 修改模型返回元组
class FixedModel(torch.nn.Module):
    def forward(self, x):
        features = self.backbone(x)
        detections = self.head(features)
        return features, detections, detections[:, :, 4]

# 然后在 dynamic_axes 中指定多个输出的维度
model = FixedModel()
dynamic_axes = {
    'input': {0: 'batch', 2: 'height', 3: 'width'},
    'features': {0: 'batch', 1: 'channels', 2: 'fmap_h', 3: 'fmap_w'},
    'detections': {0: 'batch', 1: 'num_dets'},
    'scores': {0: 'batch', 1: 'num_dets'}
}
```

### 技巧3: 符号化维度命名（Symbolic Shapes）

使用有意义的维度名称提升可读性：

```python
import torch
import onnx

class VocabModel(torch.nn.Module):
    def forward(self, tokens, mask):
        # tokens: [batch, seq_len]
        # mask: [batch, seq_len]
        return self.encoder(tokens, mask)

model = VocabModel()

# 使用符号化维度名称
dynamic_axes = {
    'tokens': {0: 'batch_size', 1: 'sequence_length'},
    'mask': {0: 'batch_size', 1: 'sequence_length'},
    'output': {0: 'batch_size', 1: 'sequence_length'}
}

dummy_input = (
    torch.randint(0, 1000, (4, 128)),  # batch=4, seq_len=128
    torch.ones(4, 128, dtype=torch.bool)
)

torch.onnx.export(
    model,
    dummy_input,
    "symbolic_dynamic.onnx",
    dynamic_axes=dynamic_axes,
    opset_version=13
)

# 检查生成的符号维度
model_onnx = onnx.load("symbolic_dynamic.onnx")
for inp in model_onnx.graph.input:
    print(f"输入: {inp.name}")
    for dim in inp.type.tensor_type.shape.dim:
        if dim.dim_param:  # 符号维度
            print(f"  维度 {dim.dim_param}: {dim.dim_value if dim.dim_value else '动态'}")
        else:
            print(f"  固定维度: {dim.dim_value}")

# 输出示例:
# 输入: tokens
#   维度 batch_size: 动态
#   维度 sequence_length: 动态
```

### 技巧4: 部分动态维度（混合固定+动态）

```python
class ImageModel(torch.nn.Module):
    def forward(self, x):
        # 假设通道数固定为 3，空间尺寸可变
        return self.extractor(x)

model = ImageModel()

# 仅高度和宽度动态，批次固定为 1
dynamic_axes = {
    'input': {2: 'height', 3: 'width'},
    'output': {2: 'height', 3: 'width'}
}

# 注意：dummy_input 仍然使用固定尺寸 (1,3,640,640)
dummy_input = torch.randn(1, 3, 640, 640)
torch.onnx.export(
    model,
    dummy_input,
    "partial_dynamic.onnx",
    dynamic_axes=dynamic_axes,
    opset_version=13
)

print("✅ 导出成功：仅 H/W 维度动态")
```

## "Fixed size required" 错误处理

### 错误示例

```python
import onnxruntime as ort
import numpy as np

# 尝试推理非标准尺寸
sess = ort.InferenceSession("model.onnx")
input_name = sess.get_inputs()[0].name
input_shape = sess.get_inputs()[0].shape

print(f"期望的输入形状: {input_shape}")
# 输出: ['batch', 3, 224, 224] 或 [1, 3, 224, 224]

# 尝试使用 480x480 推理
try:
    wrong_input = np.random.randn(1, 3, 480, 480).astype(np.float32)
    result = sess.run(None, {input_name: wrong_input})
except Exception as e:
    print(f"❌ 推理失败: {e}")
    # 典型错误: "InvalidArgument: [shape_checker] shape mismatch..."
```

### 诊断步骤

```python
def diagnose_fixed_size_error(model_path):
    """诊断固定尺寸错误的详细步骤"""
    import onnx
    import onnxruntime as ort

    # 步骤1: 检查 ONNX 模型中是否有动态维度
    model = onnx.load(model_path)
    print("=== ONNX 模型检查 ===")
    for inp in model.graph.input:
        print(f"输入 '{inp.name}' 的形状:")
        for i, dim in enumerate(inp.type.tensor_type.shape.dim):
            if dim.dim_param:
                print(f"  维度 {i}: {dim.dim_param} (动态)")
            else:
                print(f"  维度 {i}: {dim.dim_value} (固定)")
        has_dynamic = any(dim.dim_param for dim in inp.type.tensor_type.shape.dim)
        print(f">>> 是否动态: {'是' if has_dynamic else '否'}")

    # 步骤2: 检查 ONNX Runtime 的输入信息
    print("\n=== ONNX Runtime 信息 ===")
    sess = ort.InferenceSession(model_path)
    for inp in sess.get_inputs():
        print(f"输入 '{inp.name}':")
        print(f"  形状: {inp.shape}")
        print(f"  类型: {inp.type}")
        # shape 中的 'dim' 表示动态维度
        has_dynamic_ort = any(isinstance(dim, str) or dim == -1 for dim in inp.shape)
        print(f">>> 是否动态: {'是' if has_dynamic_ort else '否'}")

diagnose_fixed_size_error("model.onnx")
```

### 解决方案

#### 方案A: 重新导出为动态维度

```python
import torch

def reexport_to_dynamic(original_model_path, output_path, dynamic_axes_config):
    """重新导出模型为动态维度"""
    # 加载原始 PyTorch 模型（需要源代码）
    model = MyModel()
    model.load_state_dict(torch.load("model.pth"))
    model.eval()

    # 准备新的 dynamic_axes
    dynamic_axes = dynamic_axes_config

    # 使用新配置重新导出
    dummy_input = torch.randn(1, 3, 640, 640)  # 可以是任意尺寸
    torch.onnx.export(
        model,
        dummy_input,
        output_path,
        dynamic_axes=dynamic_axes,
        opset_version=13,
        input_names=["input"],
        output_names=["output"]
    )
    print(f"✅ 重新导出为动态维度: {output_path}")

# 配置示例：batch + H + W 动态
config = {
    'input': {0: 'batch', 2: 'height', 3: 'width'},
    'output': {0: 'batch', 2: 'height', 3: 'width'}
}
reexport_to_dynamic("model.onnx", "model_dynamic.onnx", config)
```

#### 方案B: 使用 ONNX GraphSurgeon 插入 Unsqueeze/Squeeze

```python
import onnx
import onnx_graphsurgeon as gs
import numpy as np

def make_dynamic_via_graphsurgeon(model_path, output_path):
    """使用 onnx_graphsurgeon 修改模型为动态维度"""
    # 加载模型
    model = onnx.load(model_path)
    graph = gs.import_onnx(model)

    # 查找需要修改的输出节点
    for output in graph.outputs:
        # 将输出形状中的固定维度转换为动态维度
        if output.shape is not None:
            # 将固定维度替换为 None（表示动态）
            new_shape = [dim if i == 0 else None for i, dim in enumerate(output.shape)]
            output.shape = tuple(new_shape)
            print(f"修改输出 '{output.name}': {output.shape}")

    # 清理图（移除未使用的节点）
    graph.cleanup()

    # 导出
    new_model = gs.export_onnx(graph)
    onnx.save(new_model, output_path)
    print(f"✅ 模型已修改: {output_path}")
```

## 实际案例：目标检测模型的动态维度

### 场景
YOLO 类模型需要支持：
- 动态批处理（batch=1-32）
- 动态图像尺寸（H, W 可变，如 416x416、640x640、832x832）

### 完整代码示例

```python
import torch
import torch.nn as nn

class YOLOModel(nn.Module):
    def __init__(self):
        super().__init__()
        self.backbone = nn.Sequential(
            nn.Conv2d(3, 64, 3, stride=2, padding=1),
            nn.Conv2d(64, 128, 3, stride=2, padding=1),
            # ... 更多层
        )
        self.head = nn.Conv2d(128, 255, 1)  # YOLOv5 格式

    def forward(self, x):
        features = self.backbone(x)
        output = self.head(features)
        return output

def export_yolo_dynamic():
    model = YOLOModel()
    model.eval()

    # 动态维度：batch + height + width
    dynamic_axes = {
        'input': {0: 'batch', 2: 'height', 3: 'width'},
        'output': {0: 'batch', 2: 'out_height', 3: 'out_width'}
    }

    # 使用中间尺寸导出
    dummy_input = torch.randn(1, 3, 640, 640)

    torch.onnx.export(
        model,
        dummy_input,
        "yolo_dynamic.onnx",
        dynamic_axes=dynamic_axes,
        opset_version=13,
        input_names=["input"],
        output_names=["output"],
        # 额外参数
        do_constant_folding=True,
        export_params=True
    )

    print("✅ YOLO 动态模型导出完成")

    # 验证推理
    import onnxruntime as ort
    sess = ort.InferenceSession("yolo_dynamic.onnx")

    test_cases = [
        (1, 416, 416),
        (4, 640, 640),
        (2, 832, 832),
    ]

    for batch, h, w in test_cases:
        img = np.random.randn(batch, 3, h, w).astype(np.float32)
        result = sess.run(None, {"input": img})
        print(f"  batch={batch}, {h}x{w} -> 输出形状: {result[0].shape}")

export_yolo_dynamic()
```

## 动态维度传播机制

```mermaid
graph LR
    A[导出配置<br/>dynamic_axes] --> B[ONNX 图生成]
    B --> C{检查张量形状}
    C --> D[固定维度<br/>dim.dim_value]
    C --> E[动态维度<br/>dim.dim_param]

    E --> F[符号维度名称<br/>如 "batch_size"]
    F --> G[ONNX Runtime<br/>推理时推断实际尺寸]
    G --> H[形状推断</text>]
    H --> I[动态图优化<br/>（每次推理）]

    D --> J[预计算图优化<br/>（一次性）]
    J --> K[推理时无需形状检查]

    style E fill:#ffebee
    style G fill:#e3f2fd
    style I fill:#fff3e0
```

### 动态维度对性能的影响

| 维度类型 | 编译器优化能力 | 推理首次延迟 | 推荐场景 |
|---------|--------------|-------------|---------|
| 全部固定 | 最高（全程优化） | 最低 | 生产部署，固定尺寸 |
| 1个动态维度 | 中等（部分优化） | 中等 | 可变批次/尺寸 |
| 2个动态维度 | 有限 | 较高 | 灵活推理 |
| 3+动态维度 | 最低（每次推理重新编译） | 最高 | 原型/调试 |

**性能提示**：使用 `session_options = ort.SessionOptions()` 可以启用/禁用图优化以适应动态场景。

## 高级动态: 序列模型的动态序列长度

对于 Transformer、RNN 等 NLP 模型：

```python
class TransformerModel(torch.nn.Module):
    def __init__(self, d_model, nhead, num_layers=6):
        super().__init__()
        encoder_layer = nn.TransformerEncoderLayer(d_model, nhead, batch_first=True)
        self.transformer = nn.TransformerEncoder(encoder_layer, num_layers)

    def forward(self, src, src_mask=None):
        # src: [batch, seq_len, d_model]
        output = self.transformer(src, src_key_padding_mask=src_mask)
        return output

def export_transformer_dynamic():
    model = TransformerModel(d_model=768, nhead=12)
    model.eval()

    # 动态序列长度
    dynamic_axes = {
        'src': {0: 'batch', 1: 'seq_len'},
        'src_mask': {0: 'batch', 1: 'seq_len'},
        'output': {0: 'batch', 1: 'seq_len'}
    }

    # dummy input
    dummy_src = torch.randn(2, 128, 768)  # batch=2, seq_len=128
    dummy_mask = torch.zeros(2, 128, dtype=torch.bool)

    torch.onnx.export(
        model,
        (dummy_src, dummy_mask),
        "transformer_dynamic.onnx",
        dynamic_axes=dynamic_axes,
        opset_version=13,
        input_names=["src", "src_mask"],
        output_names=["output"]
    )

    print("✅ 动态序列长度导出成功")

    # 测试推理
    import onnxruntime as ort
    sess = ort.InferenceSession("transformer_dynamic.onnx")

    test_lengths = [32, 64, 128, 256, 512]
    for seq_len in test_lengths:
        src = np.random.randn(1, seq_len, 768).astype(np.float32)
        mask = np.zeros((1, seq_len), dtype=bool)
        result = sess.run(None, {"src": src, "src_mask": mask})
        print(f"  seq_len={seq_len} -> 输出形状: {result[0].shape}")

export_transformer_dynamic()
```

## 常见错误与解决

### 错误1: "Dynamic axes not allowed for this operator"

```python
# 某些算子不支持动态维度
class BadModel(torch.nn.Module):
    def forward(self, x):
        # NMS 算子要求固定输入尺寸
        return torch.nms(x, 0.5)  # 这会导致错误

# 解决方案：拆分为两个路径
# 1. 动态导出不含 NMS 的主干/颈部
# 2. 静态导出 NMS 部分或使用 onnxruntime-extensions
```

### 错误2: "Shape inference failed"

```python
# 可能是 missing initializers 或无效的 axes 配置

# 调试:
import onnx
try:
    onnx.check_model("model.onnx")
    onnx.shape_inference.infer_shapes_path("model.onnx")
except Exception as e:
    print(f"形状推断失败: {e}")

# 解决方案:
# - 确保 dummy_input 使用合理的尺寸（非极值）
# - 检查 dynamic_axes 的索引是否在形状维度范围内
# - 使用 opset 13+ 以获得更好的形状推断
```

### 错误3: "onnxruntime error: Invalid graph"

```python
# 可能使用了不兼容的 opset 和 dynamic_axes
torch.onnx.export(
    model,
    dummy_input,
    "model.onnx",
    dynamic_axes=dynamic_axes,
    opset_version=9,  # ❌ opset 9 对 dynamic_axes 支持有限
)

# ✅ 使用 opset 11+（推荐 13 或 14）
torch.onnx.export(
    model,
    dummy_input,
    "model.onnx",
    dynamic_axes=dynamic_axes,
    opset_version=13,  # ✅ 支持完整 dynamic_axes
)
```

## 技巧汇总

1. **最小化动态维度**: 仅标记真正需要的维度，保持其他维度固定以获得更好的优化
2. **使用符号名**: `"batch_size"` 比 `"0"` 更易理解和调试
3. **保持一致性**: 所有相关输入/输出使用相同的动态轴命名
4. **测试所有场景**: 推理时验证最小、最大和典型尺寸
5. **注意 opset**: dynamic_axes 需要 opset ≥ 11，建议使用 13 或 14
6. **组合使用静态导出**: 对计算密集型部分导出为静态，对灵活部分使用动态
