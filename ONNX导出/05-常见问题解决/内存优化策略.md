# 内存优化策略

## 概述

大型模型（尤其是数十亿参数的模型）在 ONNX 转换和推理过程中面临严峻的内存挑战。本指南从转换时内存、文件大小、推理时内存三个维度提供系统的优化方案。

**标签**: #memory-optimization #large-models #efficiency

## 转换时内存管理

### 问题: 大模型导出 OOM

```python
import torch

class LargeModel(torch.nn.Module):
    def __init__(self):
        super().__init__()
        # 一个 10B 参数模型约占用 40GB (float32)
        self.layers = nn.Sequential(*[nn.Linear(7680, 7680) for _ in range(100)])

model = LargeModel()
model.eval()
dummy_input = torch.randn(1, 7680)

# ❌ 直接导出可能 OOM
torch.onnx.export(
    model,
    dummy_input,
    "large_model.onnx",  # OOM: Cannot allocate memory
    opset_version=13
)
```

### 解决方案1: 增量导出（Subgraph Export）

### 优化2: 移除未使用的初始器和节点

```python
import onnx
from onnx import helper, numpy_helper

def remove_unused_initializers(model_path, output_path):
    """移除模型中未使用的权重张量"""
    model = onnx.load(model_path)
    graph = model.graph

    # 收集所有被引用的初始器名称
    referenced_initializers = set()
    for node in graph.node:
        for inp in node.input:
            referenced_initializers.add(inp)

    # 找出未使用的初始器
    unused_initializers = []
    for initializer in graph.initializer:
        if initializer.name not in referenced_initializers:
            unused_initializers.append(initializer)
            print(f"发现未使用初始器: {initializer.name}")

    # 移除
    for initializer in unused_initializers:
        graph.initializer.remove(initializer)

    onnx.save(model, output_path)
    print(f"✅ 保存优化模型，移除 {len(unused_initializers)} 个未使用权重")

remove_unused_initializers("model.onnx", "model_stripped.onnx")
```

**典型节省**: 移除未使用权重可减少 5-30% 文件大小（取决于训练设置）。

### 优化4: 权重压缩（半精度、INT8 量化）

#### FP16 半精度量化

```python
def convert_onnx_to_fp16(model_path, output_path):
    """将 ONNX 模型转换为半精度（FP16）"""
    import onnx
    from onnx import numpy_helper

    model = onnx.load(model_path)

    # 转换所有初始器为 float16
    for initializer in model.graph.initializer:
        if initializer.data_type == onnx.TensorProto.FLOAT:
            array = numpy_helper.to_array(initializer).astype(np.float16)
            initializer.data_type = onnx.TensorProto.FLOAT16
            initializer.raw_data = array.tobytes()
            initializer.dims[:] = array.shape
            print(f"转换 {initializer.name} -> FLOAT16")

    # 检查并修正计算精度
    for node in model.graph.node:
        # 某些算子需要保持 float32（如 LayerNorm）
        if node.op_type in ['LayerNormalization', 'InstanceNormalization']:
            # 确保输入至少有一个是 float32
            pass

    onnx.save(model, output_path)
    print(f"✅ 半精度模型保存到 {output_path}")

    # 文件大小对比
    import os
    size_fp32 = os.path.getsize(model_path)
    size_fp16 = os.path.getsize(output_path)
    print(f"FP32: {size_fp32 / 1024**2:.1f} MB")
    print(f"FP16: {size_fp16 / 1024**2:.1f} MB")
    print(f"压缩率: {size_fp16/size_fp32*100:.1f}%")

convert_onnx_to_fp16("model_fp32.onnx", "model_fp16.onnx")
```

**典型收益**: 50% 文件大小减少，2倍推理加速（GPU）。

## 推理时内存优化

### 策略1: 动态批处理调整

```python
import onnxruntime as ort
import numpy as np

class DynamicBatchRunner:
    def __init__(self, model_path, max_batch=32):
        self.sess = ort.InferenceSession(
            model_path,
            providers=['CUDAExecutionProvider'] if ort.get_device() == 'GPU' else ['CPUExecutionProvider']
        )
        self.input_name = self.sess.get_inputs()[0].name
        self.max_batch = max_batch

    def run(self, inputs, batch_size):
        """根据可用内存调整批处理大小"""
        if batch_size > self.max_batch:
            print(f"警告: 批次 {batch_size} > 最大批次 {self.max_batch}")
            return self._run_in_chunks(inputs, self.max_batch)
        else:
            return self.sess.run(None, {self.input_name: inputs})

    def _run_in_chunks(self, inputs, chunk_size):
        """分块处理大批次"""
        results = []
        for i in range(0, len(inputs), chunk_size):
            chunk = inputs[i:i+chunk_size]
            result = self.sess.run(None, {self.input_name: chunk})
            results.append(result[0])
        return [np.concatenate(results, axis=0)]

# 使用
runner = DynamicBatchRunner("model.onnx", max_batch=16)

# 自动计算最大可处理批次
def find_max_batch(model_path, input_shape, device_memory_gb=8):
    """估算最大批次大小"""
    import psutil

    # 估算每张图像的内存占用
    sample_input = np.random.randn(1, *input_shape).astype(np.float32)
    sess = ort.InferenceSession(model_path)
    result = sess.run(None, {sess.get_inputs()[0].name: sample_input})

    # 简单估算（实际更复杂）
    input_memory = 1 * np.prod(input_shape) * 4  # float32 = 4 bytes
    output_memory = 1 * np.prod(result[0].shape) * 4

    per_sample_memory = input_memory + output_memory
    max_batch = int(device_memory_gb * 1024**3 / per_sample_memory * 0.7)  # 70% 利用率

    print(f"估算最大批次: {max_batch} (输入: {input_memory/1024**2:.1f}MB/样本)")
    return max_batch
```

### 策略3: 使用 IOBinding 零拷贝传输（GPU→GPU）

```python
import onnxruntime as ort
import torch

class ZeroCopyRunner:
    def __init__(self, model_path):
        self.sess = ort.InferenceSession(
            model_path,
            providers=['CUDAExecutionProvider']
        )
        self.io_binding = self.sess.io_binding()

    def run(self, torch_tensor):
        """零拷贝：PyTorch GPU Tensor 直接绑定到 ONNX Runtime"""
        # 绑定输入（共享内存）
        self.io_binding.bind_cpu_input(
            self.sess.get_inputs()[0].name,
            torch_tensor.cpu().numpy()  # 注意：仍需要 cpu→gpu 拷贝
            # 对于 GPU 输入，可以使用 bind_ortvalue_input（高级）
        )

        # 运行
        self.sess.run_with_iobinding(self.io_binding)

        # 获取输出
        output = self.io_binding.copy_outputs_to_cpu()[0]
        return output
```

import torch
from typing import Dict

class MemoryTracker:
    def __init__(self):
        self.memory_snapshots = []

    def snapshot(self, tag=""):
        """记录当前内存状态"""
        allocated = torch.cuda.memory_allocated() if torch.cuda.is_available() else 0
        reserved = torch.cuda.memory_reserved() if torch.cuda.is_available() else 0
        self.memory_snapshots.append({
            'tag': tag,
            'allocated_mb': allocated / 1024**2,
            'reserved_mb': reserved / 1024**2
        })

    def report(self):
        """生成内存报告"""
        print("=== 内存使用报告 ===")
        for snap in self.memory_snapshots:
            print(f"{snap['tag']}:")
            print(f"  已分配: {snap['allocated_mb']:.1f} MB")
            print(f"  保留: {snap['reserved_mb']:.1f} MB")
            print()

tracker = MemoryTracker()

# 在导出流程中插入监控点

## 内存优化检查表

### 转换前
- [ ] 使用 `low_cpu_mem_usage=True` 加载大模型
- [ ] 将模型和张量移回 CPU
- [ ] 设置 `torch.no_grad()`
- [ ] 清理未使用的变量和 cache（`torch.cuda.empty_cache()`）

### 导出时
- [ ] 选择合适 opset（13+）
- [ ] 启用常量折叠（`do_constant_folding=True`）
- [ ] 对于超大模型使用外部数据格式
- [ ] 导出后运行 `onnx.check_model`

### 导出后优化
- [ ] 移除未使用的初始器
- [ ] 层融合（BatchNorm、Conv-Bias）
- [ ] 权重量化（FP16、INT8）
- [ ] 形状推断
- [ ] 使用 onnxsim 简化

### 推理优化
- [ ] 配置 `ExecutionMode.ORT_SEQUENTIAL`（内存敏感）
- [ ] 使用动态批处理
- [ ] 启用内存池（PyShared）复用
- [ ] 边缘设备考虑使用 INT8 或 FP16

## 实用脚本

### 一键优化脚本

```python
#!/usr/bin/env python3
"""
ONNX 内存优化流水线
"""

import argparse
import os
import onnx
from onnxruntime.quantization import quantize_dynamic, QuantType

def optimize_pipeline(
    input_model,
    output_model,
    enable_fp16=True,
    enable_int8=False,
    enable_fusion=True
):
    """完整的 ONNX 优化流程"""
    print(f"加载模型: {input_model}")
    model = onnx.load(input_model)

    # 步骤1: 移除未使用权重
    print("1/4 移除未使用权重...")
    remove_unused_initializers(model)

    # 步骤2: 常量折叠
    print("2/4 常量折叠...")
    from onnx import optimizer
    optimizations = ["eliminate_identity"]
    if enable_fusion:
        optimizations.extend(["fuse_bn_into_conv", "fuse_add_bias_into_conv"])
    model = optimizer.optimize(model, optimizations)

    # 保存中间结果
    temp_model = "temp_optimized.onnx"
    onnx.save(model, temp_model)

    # 步骤3: 量化（可选）
    if enable_fp16:
        print("3/4 转换为 FP16...")
        convert_onnx_to_fp16(temp_model, output_model)
        os.remove(temp_model)
    elif enable_int8:
        print("3/4 动态 INT8 量化...")
        quantize_dynamic(
            model_input=temp_model,
            model_output=output_model,
            weight_type=QuantType.QInt8,
            per_channel=True
        )
        os.remove(temp_model)
    else:
        print("3/4 跳过量化")
        onnx.save(model, output_model)

    # 步骤4: 验证
    print("4/4 验证模型...")
    onnx.check(output_model)

    # 报告
    input_size = os.path.getsize(input_model) / 1024**2
    output_size = os.path.getsize(output_model) / 1024**2
    print(f"\n✅ 优化完成!")
    print(f"  输入: {input_size:.1f} MB")
    print(f"  输出: {output_size:.1f} MB")
    print(f"  压缩率: {output_size/input_size*100:.1f}%")

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--input", required=True)
    parser.add_argument("--output", required=True)
    parser.add_argument("--fp16", action="store_true")
    parser.add_argument("--int8", action="store_true")
    args = parser.parse_args()

    optimize_pipeline(args.input, args.output, args.fp16, args.int8)
```
