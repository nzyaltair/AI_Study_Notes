# 数据类型匹配问题

## 概述

ONNX（Open Neural Network Exchange）作为跨框架的模型交换格式，在模型转换过程中经常遇到数据类型不匹配的问题。不同深度学习框架使用不同的默认数据类型，这可能导致ONNX Runtime推理时出现类型错误。

**标签**: #data-types #type-mismatch #debugging

---

## 常见类型不匹配场景

### 1. 浮点数精度问题

#### PyTorch float64 vs ONNX/ONNX Runtime float32

```python
# 问题代码示例
import torch
import torch.onnx

class Model(torch.nn.Module):
    def forward(self, x):
        return x * 2.0  # Python float 默认是 float64

model = Model()
x = torch.randn(1, 3, 224, 224, dtype=torch.float64)  # 显式使用 float64

torch.onnx.export(
    model,
    x,
    "model.onnx",
    opset_version=13,
    input_names=["input"],
    output_names=["output"]
)
# ONNX Runtime 报错: "TypeError: Tensor data type float64 not supported"
```

**错误原因**：ONNX 和大多数推理引擎主要支持 `float32` 和 `float16`，不原生支持 `float64`。

#### 解决方案

```python
# 方案1: 强制使用 float32
import torch
import onnx

class Model(torch.nn.Module):
    def forward(self, x):
        # 使用 float32 常量
        return x * torch.tensor(2.0, dtype=torch.float32)

model = Model()
# 输入也使用 float32
x = torch.randn(1, 3, 224, 224, dtype=torch.float32)

torch.onnx.export(
    model,
    x,
    "model.onnx",
    opset_version=13,
    input_names=["input"],
    output_names=["output"]
)

# 方案2: 导出后修复（如果已有错误模型）
model_onnx = onnx.load("model.onnx")

# 遍历所有初始器和节点，检查并转换类型
for initializer in model_onnx.graph.initializer:
    if initializer.data_type == onnx.TensorProto.DOUBLE:
        # 转换为 float32
        converted = onnx.helper.make_tensor(
            name=initializer.name,
            data_type=onnx.TensorProto.FLOAT,
            dims=initializer.dims,
            vals=onnx.numpy_helper.to_array(initializer).astype('float32')
        )
        model_onnx.graph.initializer.remove(initializer)
        model_onnx.graph.initializer.append(converted)

onnx.check_model(model_onnx)
onnx.save(model_onnx, "model_fixed.onnx")
```

### 2. 整数类型问题：int64 vs int32

```python
# TensorFlow/Keras 常见问题
import tensorflow as tf

model = tf.keras.Sequential([
    tf.keras.layers.Dense(10),
    tf.keras.layers.Reshape((5, 2))
])

# TensorFlow shape 操作通常会输出 int64
# ONNX Runtime 期望 int32 作为 shape 输入

# 转换前检查
import onnx
model_onnx = onnx.load("model.onnx")
for node in model_onnx.graph.node:
    if node.op_type == "Reshape":
        print(f"Reshape node: {node.name}")
        # 检查 shape 输入的类型
        for inp in node.input:
            if inp != model_onnx.graph.input[0].name:
                print(f"  Shape input: {inp}")

# 修复：在导出时指定 opset 和动态 shape
```

**解决方案**：在 TensorFlow 导出时指定 `opset_version=13` 并在转换后使用 `onnxsim` 简化。

## ONNX 数据类型映射表

| 框架类型 | ONNX 类型 | TensorProto 值 | 备注 |
|---------|----------|---------------|------|
| `float32` / `tf.float32` | `FLOAT` | 1 | 最常用，完全支持 |
| `float16` / `tf.float16` | `FLOAT16` | 10 | 需要 GPU 或扩展支持 |
| `float64` / `tf.float64` | `DOUBLE` | 11 | **ONNX Runtime 不支持** |
| `int32` / `tf.int32` | `INT32` | 6 | 支持 |
| `int64` / `tf.int64` | `INT64` | 7 | 仅部分 ops 支持 |
| `int16` / `tf.int16` | `INT16` | 3 | 有限支持 |
| `int8` / `tf.int8` | `INT8` | 2 | 量化常用 |
| `uint8` / `tf.uint8` | `UINT8` | 2 | 与 int8 共享存储 |
| `bool` / `tf.bool` | `BOOL` | 9 | 支持 |
| `string` | `STRING` | 8 | 仅特定 ops |

**重要提示**：ONNX Runtime 官方支持的输入/输出类型：
- **FP32**: `FLOAT` (1) - 完全支持，推荐
- **FP16**: `FLOAT16` (10) - GPU 支持
- **INT32**: `INT32` (6) - 支持
- **INT64**: ⚠️ 仅部分 ops 支持，应避免
- **INT8/UINT8**: 量化支持

## 诊断类型问题

### 方法1: 使用 onnx.helper 检查图结构

```python
import onnx

def inspect_onnx_types(model_path):
    """检查 ONNX 模型中所有张量的数据类型"""
    model = onnx.load(model_path)
    print(f"输入类型:")
    for inp in model.graph.input:
        dtype = inp.type.tensor_type.elem_type
        print(f"  {inp.name}: {onnx.TensorProto.DataType.Name(dtype)}")
        if inp.type.tensor_type.shape.dim:
            shapes = [dim.dim_value for dim in inp.type.tensor_type.shape.dim]
            print(f"    形状: {shapes}")

    print(f"\n输出类型:")
    for out in model.graph.output:
        dtype = out.type.tensor_type.elem_type
        print(f"  {out.name}: {onnx.TensorProto.DataType.Name(dtype)}")
        shapes = [dim.dim_value for dim in out.type.tensor_type.shape.dim]
        print(f"    形状: {shapes}")

    print(f"\n中间节点数据类型检查:")
    for initializer in model.graph.initializer:
        dtype = initializer.data_type
        print(f"  {initializer.name}: {onnx.TensorProto.DataType.Name(dtype)}")

    return model

# 使用示例
model = inspect_onnx_types("model.onnx")

# 检查是否存在不支持的 DOUBLE 类型
unsupported_types = []
for initializer in model.graph.initializer:
    if initializer.data_type == onnx.TensorProto.DOUBLE:
        unsupported_types.append(initializer.name)

if unsupported_types:
    print(f"\n⚠️ 发现不支持的 DOUBLE 类型:")
    for name in unsupported_types:
        print(f"  - {name}")
```

### 方法2: 使用 onnxruntime 测试推理

```python
import onnxruntime as ort
import numpy as np

def test_onnx_type_compatibility(model_path, input_data):
    """测试 ONNX 模型是否能正确处理输入类型"""
    try:
        sess = ort.InferenceSession(model_path)
        input_name = sess.get_inputs()[0].name
        output_name = sess.get_outputs()[0].name

        result = sess.run([output_name], {input_name: input_data})
        print(f"✅ 推理成功，输出形状: {result[0].shape}")
        print(f"   输入类型: {input_data.dtype}")
        print(f"   输出类型: {result[0].dtype}")
        return True
    except Exception as e:
        print(f"❌ 推理失败: {e}")
        return False

# 测试 float32 输入
input_fp32 = np.random.randn(1, 3, 224, 224).astype(np.float32)
test_onnx_type_compatibility("model.onnx", input_fp32)

# 如果模型接受 float64，会失败
input_fp64 = np.random.randn(1, 3, 224, 224).astype(np.float64)
test_onnx_type_compatibility("model.onnx", input_fp64)
```

## 类型修复技术

### 技术1: 导出时强制类型

```python
import torch

def export_with_type_check(model, dummy_input, output_path, target_dtype=torch.float32):
    """导出模型时强制执行类型检查"""
    # 确保模型和输入使用相同类型
    model = model.to(target_dtype)
    dummy_input = dummy_input.to(target_dtype)

    # 验证前向传播
    with torch.no_grad():
        output = model(dummy_input)
        assert output.dtype == target_dtype, f"输出类型 {output.dtype} 与目标类型 {target_dtype} 不匹配"

    # 导出
    torch.onnx.export(
        model,
        dummy_input,
        output_path,
        opset_version=13,
        input_names=["input"],
        output_names=["output"],
        # 关键参数：确保运算符保持精度
        operator_export_type=torch.onnx.OperatorExportTypes.ONNX,
    )
    print(f"✅ 已导出到 {output_path}, 类型: {target_dtype}")

# 使用示例
model = MyModel()
dummy_input = torch.randn(1, 3, 224, 224, dtype=torch.float32)
export_with_type_check(model, dummy_input, "model_fp32.onnx", torch.float32)
```

### 技术2: 使用自定义转换函数

```python
import numpy as np
import onnx
from onnx import helper, numpy_helper

def cast_initializers_to_float32(model_path, output_path):
    """将模型中所有初始器转换为 float32"""
    model = onnx.load(model_path)

    for initializer in model.graph.initializer:
        if initializer.data_type != onnx.TensorProto.FLOAT:
            # 读取原有数据
            old_array = numpy_helper.to_array(initializer)
            # 转换为 float32
            new_array = old_array.astype(np.float32)

            # 创建新的初始器
            new_initializer = helper.make_tensor(
                name=initializer.name,
                data_type=onnx.TensorProto.FLOAT,
                dims=list(initializer.dims),
                vals=new_array
            )
            # 替换
            model.graph.initializer.remove(initializer)
            model.graph.initializer.append(new_initializer)

            print(f"转换 {initializer.name}: {onnx.TensorProto.DataType.Name(initializer.data_type)} -> FLOAT")

    # 验证模型
    onnx.check_model(model)
    onnx.save(model, output_path)
    print(f"✅ 已保存转换后的模型到 {output_path}")

# 使用
cast_initializers_to_float32("model.onnx", "model_fp32.onnx")
```

### 技术3: 插入 Cast 节点

```python
import onnx
from onnx import helper, numpy_helper

def insert_cast_nodes(model_path, output_path, target_type=onnx.TensorProto.FLOAT):
    """在模型输入/输出处插入 Cast 节点"""
    model = onnx.load(model_path)

    # 创建新的图节点列表
    new_nodes = list(model.graph.node)

    # 处理输入
    for input_info in model.graph.input:
        if input_info.type.tensor_type.elem_type != target_type:
            print(f"输入 {input_info.name} 需要类型转换")

            # 创建 Cast 节点
            cast_node = helper.make_node(
                "Cast",
                inputs=[input_info.name],
                outputs=[f"{input_info.name}_casted"],
                name=f"Cast_{input_info.name}"
            )
            # 设置 to 属性
            cast_node.attribute.extend([
                helper.make_attribute("to", target_type)
            ])

            new_nodes.append(cast_node)
            # 更新输入名称（实际逻辑需要重连所有引用此输入的网络）

    # 保存修改后的模型
    onnx.save(model, output_path)
    print(f"✅ 已插入 Cast 节点并保存到 {output_path}")
```

## 框架特定的类型处理

### PyTorch

```python
import torch

# 1. 导出前统一类型
def prepare_torch_model_for_onnx(model, dummy_input):
    """准备 PyTorch 模型用于 ONNX 导出"""
    model.eval()  # 切换到评估模式

    # 确保使用 float32
    model = model.float()
    dummy_input = dummy_input.float()

    # 测试推理
    with torch.no_grad():
        output = model(dummy_input)
        print(f"输出类型: {output.dtype}")

    return model, dummy_input

# 2. 使用动态类型检查装饰器
def onnx_export_safe(func):
    """装饰器：确保函数返回 float32"""
    def wrapper(*args, **kwargs):
        result = func(*args, **kwargs)
        if isinstance(result, torch.Tensor) and result.dtype != torch.float32:
            print(f"警告: 返回类型为 {result.dtype}，将转换为 float32")
            return result.float()
        return result
    return wrapper

class SafeModel(torch.nn.Module):
    @onnx_export_safe
    def forward(self, x):
        # 你的计算逻辑
        return x * 2.0
```

### TensorFlow/Keras

```python
import tensorflow as tf
import onnx
import onnxruntime as ort

# 1. 配置 TF 类型
def prepare_tf_model_for_onnx(model):
    """确保 TensorFlow 模型使用 float32"""
    # 复制模型以避免修改原模型
    model_copy = tf.keras.models.clone_model(model)
    model_copy.set_weights(model.get_weights())

    # 重建模型强制 float32
    inputs = tf.keras.Input(shape=model.input_shape[1:], dtype=tf.float32)
    outputs = model_copy(inputs)
    model_fp32 = tf.keras.Model(inputs=inputs, outputs=outputs)

    return model_fp32

# 2. 使用 tf2onnx 转换时指定类型
import tf2onnx

model_fp32 = prepare_tf_model_for_onnx(original_model)
spec = (tf.TensorSpec((None, 224, 224, 3), tf.float32, name="input"),)
onnx_model, _ = tf2onnx.convert.from_keras(
    model_fp32,
    input_signature=spec,
    opset=13
)
onnx.save(onnx_model, "model_fp32.onnx")
```

## 最佳实践总结

| 场景 | 解决方案 | 优先级 |
|------|---------|--------|
| PyTorch float64 输出 | `model.float()` + `input.float()` | ⭐⭐⭐ |
| TF shape 为 int64 | 在 Reshape/Gather 前添加 `tf.cast(x, tf.int32)` | ⭐⭐⭐ |
| 常量类型错误 | 使用 `torch.tensor(value, dtype=torch.float32)` | ⭐⭐⭐ |
| ONNX 导出时类型错误 | 检查 opset 版本，升级到 11+ | ⭐⭐ |
| 已有模型类型错误 | 使用 ONNX GraphSurgeon 插入 Cast 节点 | ⭐⭐ |
| 混合精度模型 | 分离 float16/float32 部分，避免自动类型推断 | ⭐ |
