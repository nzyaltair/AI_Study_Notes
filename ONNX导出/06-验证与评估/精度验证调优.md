# 精度验证与调优

## 概述

ONNX模型转换可能导致精度下降，这通常由算子实现差异、数值精度转换、优化策略等因素引起。本文将系统性地介绍精度验证方法、问题根源分析、以及调优策略，帮助你在性能与精度之间找到最佳平衡。

## 精度下降的原因

### 常见原因汇总表

| 原因类别 | 具体表现 | 检测方法 | 解决方案 |
|---------|---------|---------|---------|
| **算子实现差异** | 微小的数值偏差累积 | 逐层对比输出 | 调整容差阈值或修改opset |
| **量化精度损失** | INT8模型显著低于FP32 | 比较量化前后精度 | 改进校准数据、使用QAT |
| **常量折叠偏差** | 某些融合算子产生差异 | 关闭optimizer后对比 | 关闭特定优化或手动调整 |
| **动态轴处理** | 可变batch size导致问题 | 固定batch size验证 | 明确指定static batch |
| **转换bug** | 特定输入出现异常值 | 调试失败案例 | 框架issue、绕过问题算子 |
| **随机性** | Dropout/Training模式差异 | 固定seed | runtime时设置deterministic |

### 量化相关精度损失

```python
def analyze_quantization_impact(
    fp32_model_path: str,
    quant_model_path: str,
    val_dataset
):
    """
    分析量化导致精度下降

    返回:
    - top1_accuracy_fp32
    - top1_accuracy_quant
    - 每个类别的精度变化
    """
    import onnxruntime as ort
    from sklearn.metrics import classification_report

    # 加载两个模型
    fp32_sess = ort.InferenceSession(fp32_model_path)
    quant_sess = ort.InferenceSession(quant_model_path)

    # 收集预测结果
    fp32_preds = []
    quant_preds = []
    labels = []

    for batch in val_dataset:
        inputs = {fp32_sess.get_inputs()[0].name: batch['images']}
        fp32_out = fp32_sess.run(None, inputs)[0]
        quant_out = quant_sess.run(None, inputs)[0]

        fp32_preds.append(np.argmax(fp32_out, axis=1))
        quant_preds.append(np.argmax(quant_out, axis=1))
        labels.append(batch['labels'])

    # 计算整体准确率
    fp32_preds = np.concatenate(fp32_preds)
    quant_preds = np.concatenate(quant_preds)
    labels = np.concatenate(labels)

    acc_fp32 = np.mean(fp32_preds == labels)
    acc_quant = np.mean(quant_preds == labels)

    print(f"FP32准确率: {acc_fp32:.4f}")
    print(f"量化准确率: {acc_quant:.4f}")
    print(f"精度损失: {(acc_fp32 - acc_quant)*100:.2f}%")

    # 详细分类报告
    print("\nFP32分类报告:")
    print(classification_report(labels, fp32_preds))

    print("\n量化分类报告:")
    print(classification_report(labels, quant_preds))

    # 计算差异样本
    diff_mask = fp32_preds != quant_preds
    num_diff = np.sum(diff_mask)

    print(f"\n预测不一致样本: {num_diff}/{len(labels)} ({num_diff/len(labels)*100:.1f}%)")

    # 找出受影响最大的类别
    unique_labels = np.unique(labels)
    per_class_diff = {}

    for label in unique_labels:
        class_mask = labels == label
        class_diff = np.sum((fp32_preds != quant_preds) & class_mask)
        class_total = np.sum(class_mask)
        per_class_diff[label] = {
            'num_diff': int(class_diff),
            'total': int(class_total),
            'diff_rate': float(class_diff / class_total)
        }

    return {
        'acc_fp32': float(acc_fp32),
        'acc_quant': float(acc_quant),
        'accuracy_drop': float(acc_fp32 - acc_quant),
        'per_class_difference': per_class_diff,
        'num_different_samples': int(num_diff),
        'total_samples': len(labels)
    }
```

```python
from onnxruntime.quantization import quantize_static, QuantFormat, CalibrationMethod
import onnxruntime.tools.convert_onnx_models_to_ort as ort_converter

def static_quantization_with_calibration(
    fp32_model_path: str,
    quant_model_path: str,
    calibration_data_path: str,
    num_calibration_samples: int = 1000
):
    """
    静态量化：使用校准数据集确定激活值的量化范围

    步骤:
    1. 准备校准数据集
    2. 运行校准计算激活值范围
    3. 应用量化
    """
    import onnx

    # 1. 加载浮点模型
    model = onnx.load(fp32_model_path)

    # 2. 创建校准数据读取器
    class CalibrationDataReader:
        def __init__(self, data_path, input_name, num_samples):
            self.data_path = data_path
            self.input_name = input_name
            self.num_samples = num_samples
            self.current_idx = 0
            self.data = self.load_data()

        def load_data(self):
            """加载校准数据"""
            # 实现: 从.npy文件或图像目录加载
            print(f"加载校准数据: {self.data_path}")
            if self.data_path.endswith('.npy'):
                return np.load(self.data_path)[:self.num_samples]
            else:
                # 使用实际数据集
                return self.load_calibration_dataset()[:self.num_samples]

        def get_next(self):
            if self.current_idx >= len(self.data):
                return None
            batch = {self.input_name: self.data[self.current_idx:self.current_idx+1]}
            self.current_idx += 1
            return batch

        def rewind(self):
            self.current_idx = 0

    # 3. 执行量化
    print("开始静态量化...")

    # 获取输入名称
    sess = ort.InferenceSession(fp32_model_path)
    input_name = sess.get_inputs()[0].name

    calib_reader = CalibrationDataReader(
        calibration_data_path,
        input_name,
        num_calibration_samples
    )

    quantize_static(
        fp32_model_path,
        quant_model_path,
        calib_reader,
        quant_format=QuantFormat.QOperator,
        activation_type=QuantType.QUInt8,
        weight_type=QuantType.QInt8,
        per_channel=True,
        reduce_range=False
    )

    print(f"量化完成: {quant_model_path}")
    return quant_model_path
```

### 量化感知训练 (Quantization-Aware Training, QAT)

QAT在训练阶段模拟量化误差，通常比PTQ精度更高。

```python
def prepare_qat_model_pytorch(
    fp32_model_path: str,
    train_loader,
    val_loader,
    num_epochs: int = 5
):
    """
    PyTorch QAT流程

    步骤:
    1. 加载FP32预训练模型
    2. 插入fake quantization节点
    3. 微调模型（fine-tune）
    4. 导出为ONNX
    5. 应用量化
    """
    import torch
    import torch.nn as nn
    from torch.quantization import prepare_qat, convert

    # 1. 加载模型
    model = torch.load(fp32_model_path)
    model.eval()

    # 2. 设置量化配置
    model.qconfig = torch.quantization.get_default_qat_qconfig('fbgemm')
    # 可选配置:
    # - 'fbgemm' (x86 CPU, 默认)
    # - 'qnnpack' (ARM CPU)
    # - 'onednn' (Intel CPU)
    # - 'nnpi' (Habana)

    # 3. 准备QAT
    model_prepared = prepare_qat(model)

    # 4. 微调
    optimizer = torch.optim.SGD(model_prepared.parameters(), lr=0.001)
    criterion = nn.CrossEntropyLoss()

    print("开始QAT微调...")
    for epoch in range(num_epochs):
        model_prepared.train()
        running_loss = 0.0

        for batch_idx, (images, targets) in enumerate(train_loader):
            optimizer.zero_grad()
            outputs = model_prepared(images)
            loss = criterion(outputs, targets)
            loss.backward()
            optimizer.step()

            running_loss += loss.item()

            if batch_idx % 100 == 0:
                print(f"  Epoch {epoch+1}/{num_epochs}, "
                      f"Batch {batch_idx}, Loss: {loss.item():.4f}")

    # 5. 转换为量化模型
    model_quantized = convert(model_prepared.eval())

    # 6. 导出为ONNX
    dummy_input = torch.randn(1, 3, 224, 224)
    onnx_path = "model_qat.onnx"
    torch.onnx.export(
        model_quantized,
        dummy_input,
        onnx_path,
        export_params=True,
        opset_version=13,
        do_constant_folding=True,
        input_names=['input'],
        output_names=['output']
    )

    print(f"QAT模型已导出: {onnx_path}")

    # 7. 验证精度
    print("\n验证QAT模型精度:")
    validate_model_accuracy(model_quantized, val_loader)

    return onnx_path
```

```python
def generate_calibration_dataset(
    data_dir: str,
    preprocess_fn,
    num_samples: int = 1000
):
    """
    从图像目录生成校准数据集

    实际应用示例:
    """
    from torchvision import datasets, transforms
    from torch.utils.data import DataLoader

    # 加载数据集
    dataset = datasets.ImageFolder(
        root=data_dir,
        transform=transforms.Compose([
            transforms.Resize(256),
            transforms.CenterCrop(224),
            transforms.ToTensor(),
            transforms.Normalize(
                mean=[0.485, 0.456, 0.406],
                std=[0.229, 0.224, 0.225]
            )
        ])
    )

    # 随机采样
    indices = np.random.choice(len(dataset), num_samples, replace=False)
    calibration_data = []

    for idx in indices:
        img, _ = dataset[idx]
        calibration_data.append(img.numpy()[None, ...])  # 添加batch维度

    calibration_data = np.vstack(calibration_data)
    np.save("./calibration_data.npy", calibration_data)

    return calibration_data
```

```python
def export_without_optimizations(
    pytorch_model,
    dummy_input,
    output_path: str
):
    """
    关闭所有优化，排查优化器引入的精度问题

    export(
        pytorch_model,
        dummy_input,
        output_path,
        export_params=True,
        opset_version=13,
        do_constant_folding=False,  # 禁用常量折叠
        input_names=['input'],
        output_names=['output']
    )

    # 同时在ONNX Runtime session中禁用优化
    import onnxruntime as ort
    sess_options = ort.SessionOptions()
    sess_options.graph_optimization_level = ort.GraphOptimizationLevel.DISABLE_ALL

    sess = ort.InferenceSession(
        output_path,
        sess_options=sess_options
    )

    print(f"已导出并禁用优化: {output_path}")
    return output_path
```

```python
def convert_to_fp16(
    fp32_model_path: str,
    fp16_model_path: str
):
    """
    FP16量化：轻微精度损失，显著性能提升

    适用场景:
    - GPU推理（Tensor Core加速）
    - 模型本身对FP16容错性较好
    """
    import onnx
    from onnx import helper, numpy_helper

    print(f"转换为FP16: {fp32_model_path}")

    # 加载FP32模型
    model = onnx.load(fp32_model_path)

    # 转换所有浮点tensor为float16
    for tensor in model.graph.initializer:
        if tensor.data_type == onnx.TensorProto.FLOAT:
            tensor.data_type = onnx.TensorProto.FLOAT16
            tensor.raw_data = tensor.raw_data.astype(np.float16).tobytes()

    # 转换模型输入/输出类型（如果需要）
    for value_info in list(model.graph.input) + list(model.graph.output):
        if value_info.type.tensor_type.elem_type == onnx.TensorProto.FLOAT:
            value_info.type.tensor_type.elem_type = onnx.TensorProto.FLOAT16

    # 可能需要插入Cast节点进行类型转换
    # 这里简化处理，实际需要更仔细的图修改

    onnx.save(model, fp16_model_path)
    print(f"FP16模型已保存: {fp16_model_path}")

    return fp16_model_path
```

```python
def compare_calibration_methods(
    fp32_model: str,
    calibration_data,
    methods: List[str] = ['MinMax', 'Entropy', 'Percentile']
):
    """
    对比不同校准方法对精度的影响

    来源: onnxruntime.quantization.CalibrationMethod
    """
    from onnxruntime.quantization import quantize_static, CalibrationMethod

    results = {}

    for method_name in methods:
        print(f"\n校准方法: {method_name}")

        method_map = {
            'MinMax': CalibrationMethod.MinMax,
            'Entropy': CalibrationMethod.Entropy,
            'Percentile': CalibrationMethod.Percentile
        }

        quantized_path = f"model_quant_{method_name.lower()}.onnx"

        class SimpleCalibData:
            def __init__(self, data):
                self.data = data
                self.idx = 0

            def get_next(self):
                if self.idx >= len(self.data):
                    return None
                batch = {'input': self.data[self.idx:self.idx+1]}
                self.idx += 1
                return batch

            def rewind(self):
                self.idx = 0

        calib_reader = SimpleCalibData(calibration_data)

        quantize_static(
            fp32_model,
            quantized_path,
            calib_reader,
            quant_format=QuantFormat.QOperator,
            activation_type=QuantType.QUInt8,
            weight_type=QuantType.QInt8,
            calibrate_method=method_map[method_name]
        )

        # 评估精度
        acc = evaluate_model_accuracy(quantized_path, val_dataset)
        results[method_name] = {'path': quantized_path, 'accuracy': acc}

        print(f"  准确率: {acc:.4f}")

    # 推荐最佳方法
    best_method = max(results.items(), key=lambda x: x[1]['accuracy'])
    print(f"\n推荐方法: {best_method[0]} (准确率: {best_method[1]['accuracy']:.4f})")

    return results
```

## 精度-性能权衡决策指南

### 决策流程图

```
是否对精度有严格要求?
    |
    ├─ 是 (精度 > 99%原模型)
    |  ├─ 可以接受轻微精度损失?
    |  |  ├─ 2%以内 → 尝试不同opset或禁用部分优化
    |  |  ├─ 1%以内 → 使用QAT或更高bit量化（INT16）
    |  |  └─ 几乎无损 → 保持FP32或FP16混合精度
    |  |
    |  └─ 必须无损
    |     → 验证非量化方案或排查转换问题
    |
    └─ 否 (可接受5-10%精度损失)
       ├─ 资源受限 (边缘设备)
       |  → 优先选择INT8 PTQ + 优化校准数据
       |
       └─ 追求极致性能
          → 尝试INT8 QAT或探索INT4（需要工具支持）
```

### 精度损失接受度矩阵

| 任务类型 | 精度损失容忍度 | 推荐方案 | 验证指标 |
|---------|--------------|---------|---------|
| 图像分类 | < 1% top-1 | FP16 或 INT8 QAT | Top-1/Top-5 |
| 目标检测 | < 2% mAP | INT8 PTQ + 小学习率微调 | mAP@0.5 |
| 语义分割 | < 3% mIoU | INT8 PTQ + 校准数据多样性 | mIoU |
| 语音识别 | < 0.5% WER | FP16 (INT8风险高) | WER/CER |
| NLP文本分类 | < 1% 准确率 | FP16或INT8（模型鲁棒） | 准确率/F1 |
| 关键医疗诊断 | < 0.1% | 保持FP32，避免量化 | 敏感度/特异度 |

## 示例: 图像分类精度对比

### 完整精度验证脚本

```python
def comprehensive_accuracy_evaluation(
    original_model_path: str,
    quantized_model_path: str,
    val_loader,
    class_names: List[str] = None
):
    """
    完整的分类模型精度评估

    输出:
    - 混淆矩阵
    - 各类别精度对比
    - 错误案例分析
    """
    import seaborn as sns
    import matplotlib.pyplot as plt
    from sklearn.metrics import confusion_matrix, classification_report

    # 加载模型
    import onnxruntime as ort
    fp32_sess = ort.InferenceSession(original_model_path)
    int8_sess = ort.InferenceSession(quantized_model_path)

    # 收集预测
    all_fp32_preds = []
    all_int8_preds = []
    all_labels = []

    with torch.no_grad():
        for batch in val_loader:
            images, labels = batch

            # FP32推理
            fp32_out = fp32_sess.run(
                None,
                {fp32_sess.get_inputs()[0].name: images.numpy()}
            )[0]
            fp32_preds = np.argmax(fp32_out, axis=1)

            # INT8推理
            int8_out = int8_sess.run(
                None,
                {int8_sess.get_inputs()[0].name: images.numpy()}
            )[0]
            int8_preds = np.argmax(int8_out, axis=1)

            all_fp32_preds.extend(fp32_preds)
            all_int8_preds.extend(int8_preds)
            all_labels.extend(labels.numpy())

    all_fp32_preds = np.array(all_fp32_preds)
    all_int8_preds = np.array(all_int8_preds)
    all_labels = np.array(all_labels)

    # 计算整体准确率
    acc_fp32 = np.mean(all_fp32_preds == all_labels)
    acc_int8 = np.mean(all_int8_preds == all_labels)

    print("="*60)
    print("精度评估结果")
    print("="*60)
    print(f"FP32 准确率: {acc_fp32:.4f}")
    print(f"INT8 准确率: {acc_int8:.4f}")
    print(f"精度下降: {(acc_fp32 - acc_int8)*100:.2f}%")

    # 类别级对比
    print("\n各类别精度对比:")
    print(f"{'类别':<20} {'FP32':<10} {'INT8':<10} {'差异':<10}")
    print("-"*50)

    for i, class_name in enumerate(class_names or range(len(np.unique(all_labels)))):
        class_mask = all_labels == i
        if np.any(class_mask):
            fp32_acc = np.mean(all_fp32_preds[class_mask] == i)
            int8_acc = np.mean(all_int8_preds[class_mask] == i)
            diff = fp32_acc - int8_acc

            marker = "!" if diff > 0.05 else ""
            print(f"{str(class_name):<20} {fp32_acc:<10.4f} "
                  f"{int8_acc:<10.4f} {diff:<10.4f}{marker}")

    # 混淆矩阵对比
    fig, axes = plt.subplots(1, 2, figsize=(15, 6))

    cm_fp32 = confusion_matrix(all_labels, all_fp32_preds)
    cm_int8 = confusion_matrix(all_labels, all_int8_preds)

    for ax, cm, title in zip(
        axes,
        [cm_fp32, cm_int8],
        ['FP32 Confusion Matrix', 'INT8 Confusion Matrix']
    ):
        sns.heatmap(cm, annot=True, fmt='d', cmap='Blues', ax=ax)
        ax.set_title(title)
        ax.set_xlabel('Predicted')
        ax.set_ylabel('True')

    plt.tight_layout()
    plt.savefig('confusion_matrices_comparison.png', dpi=150)
    print("\n混淆矩阵图已保存: confusion_matrices_comparison.png")

    # 错误分析
    print("\n错误分析:")
    different_mask = all_fp32_preds != all_int8_preds
    num_different = np.sum(different_mask)

    print(f"量化改变预测的样本数: {num_different}/{len(all_labels)}")
    print(f"占比: {num_different/len(all_labels)*100:.1f}%")

    # 找出FP32正确但INT8出错的样本
    fp32_correct = all_fp32_preds == all_labels
    int8_wrong = all_int8_preds != all_labels
    regression_mask = fp32_correct & int8_wrong
    num_regressions = np.sum(regression_mask)

    print(f"FP32正确→INT8错误: {num_regressions} 样本")

    return {
        'acc_fp32': float(acc_fp32),
        'acc_int8': float(acc_int8),
        'accuracy_drop': float(acc_fp32 - acc_int8),
        'confusion_matrices': {
            'fp32': cm_fp32.tolist(),
            'int8': cm_int8.tolist()
        },
        'num_different_predictions': int(num_different),
        'num_regressions': int(num_regressions)
    }

# 使用示例
results = comprehensive_accuracy_evaluation(
    original_model_path="model_fp32.onnx",
    quantized_model_path="model_int8.onnx",
    val_loader=val_loader,
    class_names=['cat', 'dog', 'car', ...]
)
```

## 实用调优检查清单

### 精度问题排查清单

```
□ 步骤1: 固定随机种子，确保可复现性
□ 步骤2: 关闭所有优化器（排查优化问题）
□ 步骤3: 使用相同输入进行对比（排查数据问题）
□ 步骤4: 逐层对比中间输出（定位问题层）
□ 步骤5: 确认数据类型转换（FP32→INT8范围）
□ 步骤6: 检查校准数据质量和数量
□ 步骤7: 尝试不同opset版本
□ 步骤8: 使用QAT替代PTQ
□ 步骤9: 调整校准方法（MinMax/Entropy/Percentile）
□ 步骤10: 如精度损失过大，考虑FP16混合精度
```

### 精度恢复优先级

| 优先级 | 方法 | 复杂度 | 预期收益 |
|-------|------|--------|---------|
| P0 | 使用真实校准数据 | 低 | +2-5% |
| P0 | 增加校准样本数至500-1000 | 低 | +1-3% |
| P1 | 尝试不同校准方法 | 中 | +0.5-2% |
| P1 | 使用QAT微调 | 高 | +3-8% |
| P2 | 调整opset版本 | 低 | +0.5-1% |
| P2 | 关闭特定优化 | 低 | 恢复不过度优化 |
| P3 | 使用FP16代替INT8 | 低 | +0.5-1% |
