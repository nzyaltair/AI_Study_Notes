# 性能测试流程

## 概述

性能测试是ONNX模型优化和部署前的关键环节。本指南提供标准化的性能基准测试流程，包括延迟测量、吞吐量测试和内存分析。

## 延迟测量最佳实践

### 延迟测量核心原则

```python
import time
import numpy as np
import onnxruntime as ort
from statistics import mean, stdev, median

class LatencyBenchmark:
    def __init__(
        self,
        model_path: str,
        providers: List[str] = ['CPUExecutionProvider'],
        session_options: ort.SessionOptions = None
    ):
        """
        初始化基准测试器

        参数:
        - model_path: ONNX模型路径
        - providers: 推理后端列表
        - session_options: ONNX Runtime会话配置
        """
        self.model_path = model_path
        self.providers = providers

        if session_options is None:
            session_options = ort.SessionOptions()
            session_options.graph_optimization_level = \
                ort.GraphOptimizationLevel.ENABLE_ALL

        self.session = ort.InferenceSession(
            model_path,
            sess_options=session_options,
            providers=providers
        )

        self.input_names = [inp.name for inp in self.session.get_inputs()]
        self.output_names = [out.name for inp in self.session.get_outputs()]

    def warmup(self, num_warmup: int = 5, input_data: np.ndarray = None):
        """
        Warm-up阶段：消除首次推理的初始化开销影响

        必须执行！首次推理通常比后续推理慢3-10倍
        """
        print(f"[Warm-up] 执行 {num_warmup} 次预热推理...")

        if input_data is None:
            input_data = self.generate_random_input()

        for i in range(num_warmup):
            _ = self.session.run(self.output_names, input_data)

        print("[Warm-up] 完成")

    def measure_single_latency(
        self,
        input_data: np.ndarray,
        num_iterations: int = 100,
        skip_first: bool = True
    ) -> Dict[str, float]:
        """
        测量单次推理延迟

        返回详细的统计指标
        """
        latencies = []

        for i in range(num_iterations):
            start = time.perf_counter()
            _ = self.session.run(self.output_names, input_data)
            end = time.perf_counter()
            latencies.append((end - start) * 1000)  # 转换为ms

        # 可选择跳过第一次（已在warmup中执行）
        if skip_first and len(latencies) > 1:
            latencies = latencies[1:]

        return {
            'mean': mean(latencies),
            'median': median(latencies),
            'std': stdev(latencies) if len(latencies) > 1 else 0,
            'p50': np.percentile(latencies, 50),
            'p90': np.percentile(latencies, 90),
            'p95': np.percentile(latencies, 95),
            'p99': np.percentile(latencies, 99),
            'min': min(latencies),
            'max': max(latencies),
            'num_iterations': len(latencies)
        }

    def generate_random_input(self) -> Dict[str, np.ndarray]:
        """生成符合模型输入规格的随机数据"""
        inputs = {}
        for inp in self.session.get_inputs():
            shape = inp.shape
            # 处理动态维度
            concrete_shape = [
                dim if isinstance(dim, int) else 1
                for dim in shape
            ]
            # 使用float32作为默认数据类型
            dtype = np.float32
            inputs[inp.name] = np.random.randn(*concrete_shape).astype(dtype)
        return inputs
```

### 延迟测量完整示例

```python
def benchmark_model(
    model_path: str,
    input_shapes: Dict[str, Tuple[int, ...]],
    batch_sizes: List[int] = [1, 2, 4, 8, 16],
    num_iterations: int = 500,
    warmup_iterations: int = 10
):
    """
    综合延迟基准测试

    参数:
    - model_path: ONNX模型路径
    - input_shapes: 输入形状配置（不含batch维度）
    - batch_sizes: 测试的batch size列表
    - num_iterations: 每个配置的迭代次数
    - warmup_iterations: 预热迭代次数
    """
    results = {
        'model': model_path,
        'timestamp': time.strftime('%Y-%m-%d %H:%M:%S'),
        'batch_tests': []
    }

    for batch_size in batch_sizes:
        print(f"\n{'='*60}")
        print(f"测试 Batch Size: {batch_size}")
        print(f"{'='*60}")

        # 创建对应batch size的输入
        benchmark = LatencyBenchmark(model_path)
        test_inputs = {}

        for input_name, shape in input_shapes.items():
            concrete_shape = (batch_size,) + shape
            test_inputs[input_name] = np.random.randn(*concrete_shape).astype(np.float32)

        # 执行warmup
        benchmark.warmup(num_warmup=warmup_iterations, input_data=test_inputs)

        # 测量延迟
        latency_stats = benchmark.measure_single_latency(
            test_inputs,
            num_iterations=num_iterations
        )

        # 输出统计结果
        print(f"\n延迟统计 (n={latency_stats['num_iterations']}):")
        print(f"  平均值: {latency_stats['mean']:.2f} ms")
        print(f"  中位数: {latency_stats['median']:.2f} ms")
        print(f"  P90: {latency_stats['p90']:.2f} ms")
        print(f"  P95: {latency_stats['p95']:.2f} ms")
        print(f"  P99: {latency_stats['p99']:.2f} ms")
        print(f"  标准差: {latency_stats['std']:.2f} ms")
        print(f"  范围: [{latency_stats['min']:.2f}, {latency_stats['max']:.2f}] ms")

        results['batch_tests'].append({
            'batch_size': batch_size,
            **latency_stats
        })

    # 保存结果
    output_file = f"benchmark_latency_{Path(model_path).stem}.json"
    with open(output_file, 'w') as f:
        json.dump(results, f, indent=2)

    print(f"\n结果已保存至: {output_file}")
    return results
```

### 延迟测量注意事项

| 注意事项 | 说明 | 推荐配置 |
|---------|------|---------|
| Warm-up次数 | 首次推理包含引擎初始化开销 | CPU: 5-10次，GPU: 10-20次 |
| 迭代次数 | 统计稳定性需要足够样本 | 至少100次，P99需要500+次 |
| 采样间隔 | 避免系统干扰影响结果 | 每次迭代间隔1-2ms避免调度聚合效应 |
| 独占模式 | Linux下使用`taskset`绑定CPU核心 | `taskset -c 0-3 python benchmark.py` |
| 动态频率调节 | 禁用CPU睿频和省电模式 | BIOS中设置Performance模式 |
| 环境一致性 | 确保测试环境一致（温度、负载） | 使用专用测试机 |

### 吞吐量测试实现

```python
def measure_throughput(
    model_path: str,
    batch_size: int,
    input_shape: Tuple[int, ...],
    duration_seconds: int = 30,
    num_threads: int = 1
):
    """
    测量模型吞吐量 (samples/second)

    在指定时间内持续运行模型，统计处理的总样本数
    """
    import threading
    import queue

    # 配置多线程session（如果需要）
    sess_options = ort.SessionOptions()
    sess_options.intra_op_num_threads = num_threads
    sess_options.inter_op_num_threads = 1

    session = ort.InferenceSession(
        model_path,
        sess_options=sess_options,
        providers=['CPUExecutionProvider']
    )

    input_name = session.get_inputs()[0].name
    output_name = session.get_outputs()[0].name

    # 预生成输入批次
    test_input = np.random.randn(batch_size, *input_shape).astype(np.float32)

    # 多线程并发执行
    results_queue = queue.Queue()
    num_completed = 0
    stop_event = threading.Event()

    def worker():
        nonlocal num_completed
        while not stop_event.is_set():
            _ = session.run([output_name], {input_name: test_input})
            num_completed += batch_size
            results_queue.put(1)

    # 启动工作线程
    threads = []
    for i in range(num_threads):
        t = threading.Thread(target=worker, daemon=True)
        t.start()
        threads.append(t)

    # 运行指定时长
    time.sleep(duration_seconds)
    stop_event.set()

    # 等待所有线程完成当前推理
    for t in threads:
        t.join(timeout=1.0)

    # 计算吞吐量
    throughput = num_completed / duration_seconds

    print(f"\n吞吐量测试结果:")
    print(f"  模型: {Path(model_path).name}")
    print(f"  Batch Size: {batch_size}")
    print(f"  线程数: {num_threads}")
    print(f"  持续时间: {duration_seconds}s")
    print(f"  总样本数: {num_completed:,}")
    print(f"  吞吐量: {throughput:.2f} samples/sec")
    print(f"  等效QPS: {throughput:.2f}")

    return {
        'model': model_path,
        'batch_size': batch_size,
        'num_threads': num_threads,
        'duration_seconds': duration_seconds,
        'total_samples': num_completed,
        'throughput_samples_per_sec': throughput,
        'timestamp': time.strftime('%Y-%m-%d %H:%M:%S')
    }

def batch_size_scaling_test(
    model_path: str,
    input_shape: Tuple[int, ...],
    batch_sizes: List[int] = [1, 2, 4, 8, 16, 32, 64],
    duration: int = 10
):
    """

    results = []

    for batch_size in batch_sizes:
        try:

                input_shape=input_shape,
                duration_seconds=duration,
                num_threads=1
            )

    if results:

              f"efficiency={max_efficiency['efficiency']:.3f} sam/unit")

    return results
```

### 使用 onnxruntime-tools

```python
from onnxruntime.tools.convert_onnx_models_to_ort import convert_model
import psutil
import os

def profile_memory_usage(
    model_path: str,
    input_data: Dict[str, np.ndarray],
    num_iterations: int = 100
):
    """
    分析模型推理过程中的内存使用
    """
    import gc

    process = psutil.Process(os.getpid())

    # 初始内存
    gc.collect()
    baseline_memory = process.memory_info().rss / 1024 / 1024  # MB

    # 创建session
    session = ort.InferenceSession(model_path, providers=['CPUExecutionProvider'])

    # Session创建后的内存
    session_memory = process.memory_info().rss / 1024 / 1024
    model_load_overhead = session_memory - baseline_memory

    print(f"内存分析:")
    print(f"  基线内存: {baseline_memory:.2f} MB")
    print(f"  模型加载后: {session_memory:.2f} MB")
    print(f"  模型加载开销: {model_load_overhead:.2f} MB")

    # 预热
    for _ in range(10):
        _ = session.run(None, input_data)

    gc.collect()
    inference_baseline = process.memory_info().rss / 1024 / 1024

    peak_memory = inference_baseline

    # 执行推理并跟踪峰值内存
    for i in range(num_iterations):
        _ = session.run(None, input_data)

        current_memory = process.memory_info().rss / 1024 / 1024
        peak_memory = max(peak_memory, current_memory)

    inference_overhead = peak_memory - inference_baseline

    print(f"  推理基线内存: {inference_baseline:.2f} MB")
    print(f"  峰值内存: {peak_memory:.2f} MB")
    print(f"  推理内存开销: {inference_overhead:.2f} MB")

    return {
        'baseline_memory_mb': baseline_memory,
        'model_load_mb': model_load_overhead,
        'inference_peak_mb': inference_overhead,
        'total_peak_mb': peak_memory
    }

def memory_growth_analysis(
    model_path: str,
    input_shape: Tuple[int, ...],
    batch_sizes: List[int] = [1, 2, 4, 8, 16, 32]
):
    """
    分析内存使用随batch size的增长情况
    """
    results = []

    for batch_size in batch_sizes:
        input_data = {
            'input': np.random.randn(batch_size, *input_shape).astype(np.float32)
        }

        memory_profile = profile_memory_usage(
            model_path=model_path,
            input_data=input_data,
            num_iterations=50
        )

        results.append({
            'batch_size': batch_size,
            **memory_profile
        })

        print(f"\nBatch Size {batch_size}: "
              f"peak={memory_profile['total_peak_mb']:.1f} MB")

    return results
```

```python
class PerformanceRegression:
    """性能回归检测工具"""

    def __init__(self, baseline_file: str, current_file: str):
        self.baseline = self.load_results(baseline_file)
        self.current = self.load_results(current_file)

    def load_results(self, filepath):
        """加载性能测试结果JSON"""
        import json
        with open(filepath) as f:
            return json.load(f)

    def compare_latency(
        self,
        threshold: float = 0.1
    ) -> List[Dict]:
        """
        对比延迟变化

        参数:
        - threshold: 可接受的性能变化阈值（10%）
        返回: 性能退化列表
        """
        regressions = []

        for test in self.current.get('tests', []):
            test_name = test['name']
            baseline_test = next(
                (t for t in self.baseline.get('tests', [])
                 if t['name'] == test_name),
                None
            )

            if baseline_test:
                baseline_mean = baseline_test['mean']
                current_mean = test['mean']

                change_pct = (current_mean - baseline_mean) / baseline_mean

                if change_pct > threshold:
                    regressions.append({
                        'test': test_name,
                        'baseline_ms': baseline_mean * 1000,
                        'current_ms': current_mean * 1000,
                        'change_pct': change_pct * 100,
                        'status': 'REGRESSION'
                    })

        return regressions

    def generate_report(self, output_path: str):
        """生成性能对比报告"""
        regressions = self.compare_latency()

        report = {
            'timestamp': time.strftime('%Y-%m-%d %H:%M:%S'),
            'baseline': self.baseline.get('metadata', {}),
            'current': self.current.get('metadata', {}),
            'regressions': regressions,
            'summary': {
                'num_tests': len(self.current.get('tests', [])),
                'num_regressions': len(regressions),
                'overall_status': 'FAIL' if regressions else 'PASS'
            }
        }

        with open(output_path, 'w') as f:
            json.dump(report, f, indent=2)

        print(f"性能回归报告: {output_path}")
        print(f"退化数量: {len(regressions)} / {report['num_tests']}")
        print(f"总体状态: {report['summary']['overall_status']}")

        if regressions:
            print("\n退化详情:")
            for reg in regressions:
                print(f"  {reg['test']}: +{reg['change_pct']:.1f}% "
                      f"({reg['baseline_ms']:.2f} → {reg['current_ms']:.2f} ms)")

        return report
```

## 性能测试执行流程

### 标准测试流程

```bash
# 1. 环境准备
conda create -n onnx-benchmark python=3.9
conda activate onnx-benchmark

pip install \
  onnx \
  onnxruntime \
  onnxruntime-tools \
  pytest \
  pytest-benchmark \
  psutil

# 2. 禁用动态频率调节（Linux）
sudo cpupower frequency-set -g performance

# 3. 绑定CPU核心（可选）
taskset -c 0-3 python benchmark.py

# 4. 执行测试
python benchmark_latency.py \
  --model models/resnet50.onnx \
  --batch-sizes 1,2,4,8,16 \
  --iterations 500

# 5. 生成报告
pytest tests/test_onnx_performance.py \
  --benchmark-json results.json \
  --benchmark-compare=previous_results.json

# 6. 内存分析
python profile_memory.py --model model.onnx --batch-size 16
```

## 最佳实践

### 测试环境要求

1. **专用测试机**: 避免与其他应用竞争资源
2. **CPU亲和性**: 使用`taskset`绑定核心避免核迁移
3. **关闭频率调节**: 确保CPU以最大频率运行
4. **预热时间**: 至少10秒或50次迭代
5. **统计显著性**: 样本量 > 100，P99需要 > 500
6. **多次运行**: 至少3次完整测试取平均值
