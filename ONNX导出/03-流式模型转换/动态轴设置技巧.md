# 动态轴设置技巧

动态轴（Dynamic Axes）是ONNX转换中最强大的功能之一，它允许模型接受可变大小的输入，从而适应不同的batch size、序列长度或图像尺寸。掌握动态轴的配置技巧对于生产部署至关重要。

## dynamic_axes参数详解

`torch.onnx.export`的`dynamic_axes`参数用于指定哪些维度是动态的：

```python
torch.onnx.export(
    model,
    (input1, input2, ...),
    "model.onnx",
    input_names=['input1', 'input2'],
    output_names=['output1', 'output2'],
    dynamic_axes={
        'input1': {0: 'batch_size', 2: 'sequence_length'},
        'input2': {0: 'batch_size'},
        'output1': {0: 'batch_size', 2: 'sequence_length'},
    }
)
```

**语法说明**：
- 键：输入/输出名称
- 值：字典映射维度索引到维度名称
- 维度名称可以是任意字符串，在推理时可以替换为实际值

## 基本动态维度示例

### 1. 动态Batch Size

```python
import torch
import torch.nn as nn

class SimpleModel(nn.Module):
    def __init__(self, input_dim, output_dim):
        super().__init__()
        self.linear = nn.Linear(input_dim, output_dim)

    def forward(self, x):
        return self.linear(x)

model = SimpleModel(768, 768)
model.eval()

# 示例1：固定batch size转换为动态batch
dummy_input = torch.randn(4, 768, requires_grad=True)

torch.onnx.export(
    model,
    dummy_input,
    "dynamic_batch.onnx",
    input_names=['input'],
    output_names=['output'],
    dynamic_axes={
        'input': {0: 'batch_size'},
        'output': {0: 'batch_size'}
    },
    opset_version=13
)
```

现在模型可以接受任意batch size的输入：
```python
# Valid inputs:
# - batch_size=1: (1, 768)
# - batch_size=32: (32, 768)
# - batch_size=128: (128, 768)
```

### 2. 动态Sequence Length（NLP场景）

```python
class TransformerModel(nn.Module):
    def __init__(self, d_model=768, nhead=12, num_layers=12):
        super().__init__()
        encoder_layer = nn.TransformerEncoderLayer(
            d_model=d_model, nhead=nhead, batch_first=True
        )
        self.transformer = nn.TransformerEncoder(encoder_layer, num_layers)
        self.fc = nn.Linear(d_model, 10000)

    def forward(self, x, src_mask=None):
        x = self.transformer(x, src_mask)
        return self.fc(x.mean(dim=1))

model = TransformerModel()
model.eval()

# Batch=2, Sequence=50的示例
dummy_input = torch.randn(2, 50, 768)

torch.onnx.export(
    model,
    dummy_input,
    "dynamic_seq.onnx",
    input_names=['input_ids'],
    output_names=['logits'],
    dynamic_axes={
        'input_ids': {0: 'batch_size', 1: 'seq_length'},
        'logits': {0: 'batch_size'}
    },
    opset_version=14
)
```

### 3. 动态Height/Width（CV场景）

```python
class CNNModel(nn.Module):
    def __init__(self):
        super().__init__()
        self.conv1 = nn.Conv2d(3, 64, 3, padding=1)
        self.conv2 = nn.Conv2d(64, 128, 3, padding=1)
        self.pool = nn.AdaptiveAvgPool2d((1, 1))
        self.fc = nn.Linear(128, 10)

    def forward(self, x):
        x = self.conv1(x).relu()
        x = self.conv2(x).relu()
        x = self.pool(x)
        x = x.view(x.size(0), -1)
        return self.fc(x)

model = CNNModel()
model.eval()

dummy_input = torch.randn(4, 3, 224, 224)

torch.onnx.export(
    model,
    dummy_input,
    "dynamic_hw.onnx",
    input_names=['image'],
    output_names=['predictions'],
    dynamic_axes={
        'image': {0: 'batch_size', 2: 'height', 3: 'width'},
        'predictions': {0: 'batch_size'}
    },
    opset_version=13
)
```

## 复杂动态形状

### 多个输入输出的动态轴

```python
class MultiInputModel(nn.Module):
    def __init__(self):
        super().__init__()
        self.fc = nn.Linear(512, 256)

    def forward(self, x1, x2):
        # x1: [batch, seq1, dim]
        # x2: [batch, seq2, dim]
        out1 = self.fc(x1)
        out2 = self.fc(x2)
        return out1, out2

model = MultiInputModel()
model.eval()

dummy_x1 = torch.randn(2, 30, 512)
dummy_x2 = torch.randn(2, 50, 512)

torch.onnx.export(
    model,
    (dummy_x1, dummy_x2),
    "multi_input_dynamic.onnx",
    input_names=['input1', 'input2'],
    output_names=['output1', 'output2'],
    dynamic_axes={
        'input1': {0: 'batch', 1: 'seq1'},
        'input2': {0: 'batch', 1: 'seq2'},
        'output1': {0: 'batch', 1: 'seq1'},
        'output2': {0: 'batch', 1: 'seq2'}
    },
    opset_version=14
)
```

### 部分维度动态

并非所有维度都需要动态：

```python
# 仅动态batch size，固定序列长度和特征维度
dynamic_axes = {
    'input': {0: 'batch_size'},  # 只有batch维度动态
    'output': {0: 'batch_size'}
}

# 动态height但固定width（例如处理不同高度的图像，但宽度固定）
dynamic_axes = {
    'image': {0: 'batch', 2: 'height'},  # width固定
    'output': {0: 'batch'}
}
```

## Symbolic Shape Inference

对于更复杂的动态形状依赖，可以使用符号推导：

```python
from torch.onnx import symbolic_helper

# 自定义符号函数示例
def export_with_symbolic(model, dummy_input, output_path):
    # 定义符号维度关系
    # 例如：output_dim = input_dim + offset
    torch.onnx.export(
        model,
        dummy_input,
        output_path,
        input_names=['x'],
        output_names=['y'],
        dynamic_axes={
            'x': {0: 'batch', 1: 'seq_len'},
            'y': {0: 'batch', 1: 'seq_len'}
        },
        opset_version=15,  # 更高opset支持更复杂的符号推理
        custom_opsets=None,
        export_params=True,
        do_constant_folding=True,
        input_axes_labels=None,
        output_axes_labels=None
    )
```

### 使用symbolic trace处理控制流

```python
def symbolic_dynamic_shape_example(x):
    # 条件操作，依赖动态形状
    batch_size = x.shape[0]
    seq_len = x.shape[1]

    # 这种控制流需要特殊处理
    if seq_len > 10:
        x = x[:, :10, :]

    return x
```

对于包含控制流的模型，建议：
1. 使用`torch.jit.trace`或`torch.jit.script`先转换为TorchScript
2. 导出时使用`dynamic_axes`明确指定动态维度
3. 考虑使用`opset_version >= 14`以获得更好的动态形状支持

## 完整示例：多模态模型的动态轴配置

```python
class MultimodalModel(nn.Module):
    def __init__(self, image_encoder, text_encoder, fusion_layer):
        super().__init__()
        self.image_encoder = image_encoder
        self.text_encoder = text_encoder
        self.fusion = fusion_layer

    def forward(self, images, input_ids, attention_mask):
        # images: [batch, 3, H, W] - 动态H, W
        # input_ids: [batch, seq_len] - 动态seq_len
        img_features = self.image_encoder(images)
        text_features = self.text_encoder(input_ids, attention_mask)

        # 池化图像特征
        img_pooled = img_features.mean(dim=[2, 3])  # [batch, img_dim]

        # 文本特征池化
        text_pooled = text_features.last_hidden_state.mean(dim=1)  # [batch, text_dim]

        # 融合
        fused = self.fusion(img_pooled, text_pooled)
        return fused

# 导出时的dynamic_axes配置
dummy_images = torch.randn(2, 3, 640, 480)  # 可变分辨率
dummy_input_ids = torch.randint(0, 1000, (2, 128))
dummy_attention_mask = torch.ones(2, 128)

torch.onnx.export(
    model,
    (dummy_images, dummy_input_ids, dummy_attention_mask),
    "multimodal_dynamic.onnx",
    input_names=['images', 'input_ids', 'attention_mask'],
    output_names=['fused_features'],
    dynamic_axes={
        'images': {0: 'batch', 2: 'height', 3: 'width'},
        'input_ids': {0: 'batch', 1: 'seq_len'},
        'attention_mask': {0: 'batch', 1: 'seq_len'},
        'fused_features': {0: 'batch'}
    },
    opset_version=14
)
```

## 动态形状验证

导出后验证模型是否支持动态形状：

```python
import onnx
from onnxruntime import InferenceSession

def verify_dynamic_model(model_path):
    # 加载模型
    model = onnx.load(model_path)
    onnx.checker.check_model(model)

    # 查看输入输出形状
    print("Input Info:")
    for inp in model.graph.input:
        print(f"  {inp.name}:")
        for dim in inp.type.tensor_type.shape.dim:
            if dim.dim_value:
                print(f"    Fixed: {dim.dim_value}")
            elif dim.dim_param:
                print(f"    Dynamic: {dim.dim_param}")

    print("\nOutput Info:")
    for out in model.graph.output:
        print(f"  {out.name}:")
        for dim in out.type.tensor_type.shape.dim:
            if dim.dim_value:
                print(f"    Fixed: {dim.dim_value}")
            elif dim.dim_param:
                print(f"    Dynamic: {dim.dim_param}")

    # ONNX Runtime测试不同输入
    session = InferenceSession(model_path)

    test_cases = [
        (1, 224, 224),   # batch=1
        (8, 224, 224),   # batch=8
        (4, 384, 384),   # 不同分辨率
    ]

    for batch, h, w in test_cases:
        dummy_input = np.random.randn(batch, 3, h, w).astype(np.float32)
        try:
            outputs = session.run(None, {'image': dummy_input})
            print(f"✓ batch={batch}, H={h}, W={w} works, output shape: {outputs[0].shape}")
        except Exception as e:
            print(f"✗ batch={batch}, H={h}, W={w} failed: {e}")

verify_dynamic_model("dynamic_hw.onnx")
```

## 常见错误与调试

### 错误1：Rank mismatch
```
RuntimeError: ONNX symbolic failed for ... Input rank mismatch
```

**解决**：确保所有输入张量在导出时具有相同的rank，即使某些维度是动态的。

```python
# 错误：不同导出调用使用不同shape
torch.onnx.export(model, torch.randn(1, 10, 512), ...)  # ok
torch.onnx.export(model, torch.randn(1, 20, 512), ...)  # 可能错误

# 正确：使用2D/3D示例，明确dynamic_axes
dummy_input = torch.randn(1, 10, 512)  # seq_len只需要一个示例值
torch.onnx.export(
    model, dummy_input,
    dynamic_axes={'input': {1: 'seq_len'}},  # 指定第1维动态
    ...
)
```

### 错误2：Constant folding干扰动态shape

```python
# 关闭常量折叠以保留动态shape
torch.onnx.export(
    ...,
    do_constant_folding=False  # 对于复杂动态shape可能需要
)
```

### 错误3：Symbolic shape inference失败

```python
# 使用更简单的shape更好
dummy_input = torch.randn(2, 128, 768)  # 使用实际值而不是0或1

# 对于Transformer，确保attention mask正确
attention_mask = torch.ones(2, 128, dtype=torch.long)

torch.onnx.export(
    model,
    (dummy_input, attention_mask),
    ...,
    dynamic_axes={
        'input': {0: 'batch', 1: 'seq'},
        'attention_mask': {0: 'batch', 1: 'seq'}
    }
)
```

## 高级技巧

### 1. 嵌套动态形状

对于某些复合类型（如list/dict），需要特殊处理：

```python
class ModelWithDict(nn.Module):
    def forward(self, x):
        # 假设输出是dict
        return {
            'feature1': x.mean(dim=1),
            'feature2': x.max(dim=1).values
        }

# ONNX不支持直接输出dict，需要包装为tuple
class ModelWrapper(nn.Module):
    def forward(self, x):
        out = ModelWithDict()(x)
        return out['feature1'], out['feature2']
```

### 2. 有条件动态维度

```python
# 某些维度可能在某些条件下才会出现
# 建议：避免条件维度，使用pad或mask处理
def forward_with_padding(x, mask):
    # 统一使用最大可能长度，用mask标记真实长度
    outputs = model(x)
    masked_outputs = outputs * mask.unsqueeze(-1)
    return masked_outputs
```

### 3. 性能考虑

动态形状会影响推理性能：
- **多batch推理**：尽量使用固定的batch size以获得最佳性能
- **序列长度**：如果长度相对固定，考虑有限几个预定义长度
- **混合精度**：动态轴不影响精度选择

## 与opset版本的关系

不同opset版本对动态形状的支持：

| Opset | 特性 |
|-------|------|
| 10-11 | 基础dynamic_axes |
| 12-13 | 改进的shape推理 |
| 14-15 | 更好的控制流支持 |
| 16+ | Symbolic shape propagation增强 |

**推荐**：对于新项目，使用opset >= 14
